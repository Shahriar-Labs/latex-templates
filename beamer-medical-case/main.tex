        \item Troponin q6h until trending down
        \item Repeat echocardiogram in 24 hours
        \item Daily ECG for ST-segment evolution
      \end{itemize}
    \end{tcolorbox}

    \column{0.5\textwidth}
    \begin{tcolorbox}[infobox,title={\faUserMd\hspace{4pt}Non-Pharmacologic}]
      \small
      \begin{itemize}
        \item \textbf{Activity:} Strict bed rest with head-of-bed elevated to 30\textdegree, avoid supine positioning
        \item \textbf{Diet:} Cardiac diet (2\,g sodium restriction)
        \item \textbf{Fluids:} 1.5\,L/day restriction while effusion present
        \item \textbf{Oxygen:} Titrate to SpO\textsubscript{2} $>92$\% 
        \item \textbf{Pain:} Reassess q4h using numeric rating scale
      \end{itemize}
    \end{tcolorbox}

    \vspace{0.2cm}

    \begin{tcolorbox}[alertbox,title={\faBell\hspace{4pt}Red Flags -- Escalate Immediately}]
      \small
      \begin{itemize}
        \item Hypotension + JVD + muffled heart sounds (\textbf{Beck's triad} $\rightarrow$ tamponade)
        \item Worsening chest pain despite NSAID therapy
        \item New oxygen requirement $>4$\,L NC
      \end{itemize}
    \end{tcolorbox}
  \end{columns}
