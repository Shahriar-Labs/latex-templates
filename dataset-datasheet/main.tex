\section{Motivation \& Purpose}

\begin{motivationbox}
  \textbf{\textcolor{accent}{\faBullseye}\quad Why This Dataset Exists.}
  \smallskip

  Most multilingual text classification benchmarks are limited to 3--5 high-resource
  languages and rely on translated rather than natively-authored content.
  \textbf{MultiLingua-100K} fills this gap by providing 100{,}000 human-annotated news
  articles across \textbf{12 languages} spanning 6 language families, all sourced from
  original publications in each target language. The dataset is designed to support
  cross-lingual transfer learning, language-agnostic model evaluation, and fairness
  audits of NLP classifiers on under-represented languages.
\end{motivationbox}

\subsection{Category Distribution}
Each category is represented with at least 8{,}000 articles. The smallest category
(\textit{Science}, 8{,}420 articles) and the largest (\textit{Politics}, 22{,}100 articles)
reflect the natural imbalance observed in global news coverage. Stratified sampling
ensures fair representation across all train/val/test splits.

\subsection{Key Design Decisions}
\begin{itemize}
  \item \textbf{No machine translation.} All articles are natively authored in the target
        language to avoid translationese artifacts that degrade classifier robustness.
  \item \textbf{Minimal text alteration.} Only NFC Unicode normalization and lowercasing
        were applied. Paragraph boundaries, original letter casing, and diacritics are
        preserved in a parallel \texttt{raw\_text} field.
  \item \textbf{Multi-stage dedup.} Exact URL dedup $\rightarrow$ near-duplicate detection
        via MinHash $\rightarrow$ manual inspection of borderline pairs
        ($0.80 \le \text{sim} < 0.85$).
\end{itemize}

\begin{warnbox}
  \textbf{\textcolor{orange}{\faExclamationTriangle}\quad Coverage Gap.}
  The dataset under-represents Turkic and Koreanic language families. Turkish and Korean
  each account for fewer than 5\% of total articles. We strongly recommend stratified
  evaluation and caution against treating per-language metrics as equally reliable.
\end{warnbox}

\section{Recommended Tasks}

\subsection*{Primary (Well-Supported)}
\begin{itemize}[label={\textcolor{accent}{\faCheck}}]
  \item \textbf{Multilingual text classification} — 7-way news category prediction,
        single-label. Strong baseline: mBERT achieves 0.82 macro-F1.
  \item \textbf{Cross-lingual zero-shot transfer} — Train on English, evaluate on all 11
        target languages. XLM-R achieves 0.71 avg.\ zero-shot F1 (range: 0.58--0.84).
  \item \textbf{Language-agnostic model benchmarking} — Compare mono-, multi-, and
        language-agnostic architectures on a fixed, diverse testbed.
  \item \textbf{Fairness auditing} — Measure per-language performance disparities;
        identify systematic under-performance on low-resource splits.
\end{itemize}

\subsection*{Secondary (Exploratory)}
\begin{itemize}[label={\textcolor{midgray}{\faCircle[regular]}}]
  \item Topic modeling and cross-lingual theme discovery.
  \item Named entity recognition fine-tuning (entity annotations not included; must be
        layered independently).
  \item News source bias analysis using the \texttt{source\_id} metadata field.
  \item Temporal distribution shift analysis across the 2018--2023 span.
\end{itemize}

\section{Limitations \& Ethical Considerations}

\begin{warnbox}
  \textbf{\textcolor{orange}{\faExclamationTriangle}\quad Known Limitations.}
  \begin{itemize}[leftmargin=*]
    \item \textbf{News-domain bias.} All articles are journalistic prose; performance
          does not generalize to social media, legal, or clinical text domains.
    \item \textbf{Temporal cutoff.} Articles end at December 2023. Models trained on this
          data will not reflect post-2023 events or terminology shifts.
    \item \textbf{Annotation subjectivity.} The 7 categories are broad; edge cases
          (e.g., \textit{sports technology}) were resolved by majority vote, which may
          mask genuine semantic ambiguity.
    \item \textbf{Source selection bias.} 340 outlets skew toward major publications;
          independent and local news sources are under-represented.
  \end{itemize}
\end{warnbox}

\medskip
\textbf{Ethical review.} This dataset was reviewed by the GNLPI Ethics Board (Approval
\#GNLPI-2023-042). No personally identifiable information (PII) is included. Articles
containing hate speech, graphic violence, or adult content were excluded during
Phase~III filtering. The scraping process respected \texttt{robots.txt} and site-specific
terms of service.
