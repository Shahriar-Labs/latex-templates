\newcommand{\abilityhex}[3]{%
  \begin{tikzpicture}[baseline={(0,-0.1)}]
    % Hexagon shape
    \node (hex) [draw, regular polygon, regular polygon sides=6, 
      minimum width=1.65cm, minimum height=1.65cm,
      line width=1.2pt, charcoal, fill=parchment,
      rotate=0, inner sep=0pt] at (0,0) {};
    % Label
    \node[anchor=center, font=\bfseries\tiny\color{charcoal!80}] at (0, 0.45) {\MakeUppercase{#1}};
    % Score
    \node[anchor=center, font=\LARGE\bfseries\color{charcoal}] at (0, 0.0) {#2};
    % Modifier pill
    \node[draw=accent, fill=white, rounded corners=2pt, line width=0.8pt, 
      inner sep=1.8pt, font=\bfseries\scriptsize\color{accent}, anchor=center] at (0, -0.5) {#3};
  \end{tikzpicture}%
}
