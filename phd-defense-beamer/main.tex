\sectionslide{Part 1}{Motivation \& Problem}

% ---- motivation ----
\begin{frame}{Motivation}
  \begin{columns}[T]
    \column{0.58\textwidth}
      Deep sequence models power translation, forecasting, and clinical
      prediction --- yet they \alert{degrade silently} when deployment data drifts
      from training data.
      \begin{itemize}
        \item Failures are often \textbf{undetected} until downstream harm.
        \item Retraining is costly and \textbf{reactive}.
        \item No unified framework links \textbf{detection} to \textbf{mitigation}.
      \end{itemize}
    \column{0.42\textwidth}
      \begin{block}{The gap}
        We can measure accuracy \emph{after} labels arrive --- but need to flag
        shift \emph{before} performance collapses.
      \end{block}
  \end{columns}
\end{frame}

% ---- research questions ----
\begin{frame}{Research Questions}
  \begin{enumerate}
    \item[\textbf{RQ1}] Can we detect distribution shift from \textbf{unlabeled} model
      internals, in real time?
    \item[\textbf{RQ2}] Which representation-level signals best \textbf{predict} accuracy
      loss under shift?
    \item[\textbf{RQ3}] Can a lightweight adaptation \textbf{recover} performance without
      full retraining?
  \end{enumerate}
  \vspace{4pt}
  \begin{block}{Thesis statement}
    Representation-drift signals provide an early, label-free indicator of accuracy loss,
    and enable targeted adaptation that restores most of the lost performance.
  \end{block}
\end{frame}

% ---- contributions ----
\begin{frame}{Contributions}
  \begin{columns}[T]
    \column{0.33\textwidth}
      \begin{block}{1. Detector}
        A label-free drift score from hidden-state statistics; $O(d)$ per step.
      \end{block}
    \column{0.33\textwidth}
      \begin{block}{2. Analysis}
        A study linking drift magnitude to accuracy loss across 6 datasets.
      \end{block}
    \column{0.33\textwidth}
      \begin{block}{3. Mitigation}
        A test-time adaptation recovering 78\% of lost accuracy at 4\% compute.
      \end{block}
  \end{columns}
\end{frame}

\sectionslide{Part 2}{Method}

% ---- method ----
\begin{frame}{Method: Representation-Drift Detection}
  For hidden states $h_t$, we track the divergence between a reference window
  $\mathcal{R}$ and a live window $\mathcal{L}$:
  \begin{equation}
    D_t = \mathrm{MMD}^2\!\left(\mathcal{R},\mathcal{L}\right)
        = \left\lVert \mu_\mathcal{R} - \mu_\mathcal{L}\right\rVert^2_{\mathcal{H}}.
  \end{equation}
  A shift is flagged when $D_t$ exceeds a calibrated threshold $\tau$.
  \vspace{4pt}

% ---- analysis ----
\begin{frame}{Analysis: When Does It Work?}
  \begin{itemize}
    \item \textbf{Strong}: gradual covariate shift --- drift score rises smoothly, giving
      long lead time.
    \item \textbf{Moderate}: label shift --- detectable, but recovery depends on class
      overlap.
    \item \alert{Limitation}: adversarial or abrupt shifts leave little lead time; detection
      still fires but mitigation lags.
  \end{itemize}
  \begin{block}{Why it generalizes}
    The signal lives in \emph{representations}, not task labels --- so it transfers across
    domains and architectures.
  \end{block}
\end{frame}

      \end{itemize}
