\section{Introduction}
\label{sec:intro}

Long-range dependency modelling lies at the heart of tasks ranging from
scientific document understanding~\cite{lo2020s2orc} to genomic sequence
analysis~\cite{ji2021dnabert}. While the Transformer~\cite{vaswani2017attention}
has become the dominant architecture, its self-attention mechanism incurs
$\mathcal{O}(n^2)$ time and memory complexity with respect to sequence length $n$.

Recent efforts to mitigate this bottleneck fall into three camps:
\begin{itemize}
  \item \emph{Sparse attention patterns} (\eg~Longformer~\cite{beltagy2020longformer},
        BigBird~\cite{zaheer2020bigbird}) restrict attention to local windows plus
        global tokens.
  \item \emph{Linear attention approximations} (\eg~Performer~\cite{choromanski2020rethinking})
        replace the softmax kernel with low-rank feature maps.
  \item \emph{Recurrent hybrids} (\eg~RWKV~\cite{peng2023rwkv}) blend attention
        with recurrent dynamics.
\end{itemize}

Each strategy trades off expressiveness, hardware efficiency, or training
stability. \modelname{} takes a complementary approach: rather than altering the
attention kernel itself, we impose a \emph{dynamic information bottleneck} at the
inter-layer boundary, enabling each token to selectively draw from a compressed
but globally informed context buffer.

\subsection{Contributions}

Our main contributions are:

\begin{enumerate}[label=\textbf{\arabic*.}]
  \item We propose the \textbf{Context Gateway (CG)} layer, a differentiable
        module that routes token representations to a fixed-size context bank via
        top-$k$ selection (\S\ref{sec:method}).
  \item We provide a theoretical analysis showing that \modelname{} with $k$
        selected contexts achieves an effective attention span of
        $\mathcal{O}(kn)$ with $\mathcal{O}(kn)$ complexity (\S\ref{sec:theory}).
  \item We demonstrate consistent improvements across eight benchmarks with a
        single set of hyperparameters (\S\ref{sec:experiments}).
\end{enumerate}

\section{Related Work}
\label{sec:related}

\subsubsection{Sparse Transformers}
Correia~\etal~\cite{correia2019adaptively} propose $\alpha$-entmax, which
induces sparse attention patterns. Our gating is related but operates at the
\emph{inter-layer} level rather than within the attention head.

\subsubsection{Conditional Computation}
Mixture-of-Experts (MoE)~\cite{shazeer2017outrageously} routes tokens to
specialised feed-forward modules. \modelname{} similarly exploits conditional
computation but routes \emph{context}, not parameters.

\subsubsection{Memory-Augmented Transformers}
Memformer~\cite{wu2020memformer} and Compressive Transformer~\cite{rae2020compressive}
maintain external memory banks. Unlike these, our context bank is entirely
\emph{within-sequence} and requires no segment-level recurrence.

\section{Method}
\label{sec:method}

\subsection{Notation}

Let $\mathbf{X}^{(\ell)} \in \mathbb{R}^{n \times d}$ denote the token
representations output by encoder layer $\ell$, where $n$ is sequence length
and $d$ is the model dimension.

\subsection{Context Gateway}

The Context Gateway maps $\mathbf{X}^{(\ell)}$ to an updated representation
$\hat{\mathbf{X}}^{(\ell)}$ via three sub-operations.

\subsubsection{Context-Bank Write}
A learned linear projection $W_w \in \mathbb{R}^{d \times d_c}$ maps each token
to a candidate context vector. We then select the top-$k$ tokens by a learned
scalar score $s_i = \mathbf{w}_s^\top \mathbf{x}_i^{(\ell)}$ and write their
projections to the bank:
\begin{equation}
  C = \softmax\!\left(\frac{s_{\pi}}{\sqrt{d_c}}\right)
      \mathbf{X}^{(\ell)}_\pi W_w,
  \label{eq:write}
\end{equation}
where $\pi$ denotes the indices of the top-$k$ scoring tokens and
$C \in \mathbb{R}^{k \times d_c}$ is the context bank.

\subsubsection{Context-Bank Read}
Every token reads from the bank via scaled dot-product attention with a single
head:
\begin{equation}
  \hat{\mathbf{x}}_i^{(\ell)} =
    \mathbf{x}_i^{(\ell)} +
    \softmax\!\!\left(\frac{\mathbf{x}_i^{(\ell)} W_q\, C^\top}{\sqrt{d_c}}\right)
    C\, W_v,
  \label{eq:read}
\end{equation}
where $W_q, W_v \in \mathbb{R}^{d \times d_c}$ are query and value projections.

\subsubsection{Complexity}
Writing costs $\mathcal{O}(n)$ to compute scores and $\mathcal{O}(kd)$ to
populate the bank. Reading costs $\mathcal{O}(nk)$ per layer. Thus the
\emph{total} inter-layer overhead is $\mathcal{O}(nk + kd)$, linear in both $n$
and $k$.

\subsection{Training Objective}

We augment the task loss $\mathcal{L}_{\text{task}}$ with a sparsity regulariser
that encourages few writes without collapsing to a trivial solution:
\begin{equation}
  \mathcal{L} = \mathcal{L}_{\text{task}} +
                \lambda \sum_\ell \frac{1}{n}\sum_{i=1}^n
                \log\bigl(1 + e^{-s_i^{(\ell)}}\bigr).
  \label{eq:loss}
\end{equation}

\section{Theoretical Analysis}
\label{sec:theory}

\begin{theorem}
  \label{thm:span}
  For a \modelname{} model with $L$ encoder layers and context bank size $k$,
  each output token has an \emph{effective attention span} of at least
  $\min(n,\, k \cdot \lfloor L/2 \rfloor)$ tokens.
\end{theorem}

\begin{proof}
  By induction on layer depth. At layer $\ell=1$ each token can attend to at
  most $k$ context tokens via the bank. At layer $\ell+1$ those $k$ bank
  entries each carry information from up to $k$ distinct tokens; the bank can
  therefore represent up to $k^2$ tokens in principle, but because the bank has
  capacity $k$ entries, the span grows linearly as $k\ell$. A formal argument
  by induction over $\ell = 1, \ldots, \lfloor L/2 \rfloor$ yields the bound.
\end{proof}

\begin{corollary}
  With $k = \sqrt{n}$ and $L = \mathcal{O}(\log n)$ layers, \modelname{}
  achieves full-sequence span in $\mathcal{O}(n \log n)$ total complexity.
\end{corollary}

% ── Algorithm box ────────────────────────────────────────────
\begin{algorithm}[!t]
\DontPrintSemicolon
\SetAlgoLined
\KwIn{Token representations $\mathbf{X}^{(\ell)}$, bank size $k$}
\KwOut{Updated representations $\hat{\mathbf{X}}^{(\ell)}$}
\BlankLine
Compute scores $s_i \leftarrow \mathbf{w}_s^\top \mathbf{x}_i^{(\ell)}$, $\forall i$\;
Select top-$k$ indices $\pi \leftarrow \mathrm{argtopk}(s, k)$\;
Write bank: $C \leftarrow \softmax(s_\pi / \sqrt{d_c})\, \mathbf{X}_\pi^{(\ell)} W_w$\;
\BlankLine
\For{$i \leftarrow 1$ \KwTo $n$}{
  $a_i \leftarrow \softmax\!\!\left(\mathbf{x}_i^{(\ell)} W_q\, C^\top / \sqrt{d_c}\right)$\;
  $\hat{\mathbf{x}}_i^{(\ell)} \leftarrow \mathbf{x}_i^{(\ell)} + a_i\, C\, W_v$\;
}
\Return $\hat{\mathbf{X}}^{(\ell)}$\;
\caption{Context Gateway Forward Pass}
\label{alg:cg}
\end{algorithm}

\section{Experiments}
\label{sec:experiments}

\subsection{Datasets and Baselines}

We evaluate on eight tasks from the SCROLLS benchmark~\cite{shaham2022scrolls}:
\textsc{QuALITY}, \textsc{QASPER}, \textsc{NarrativeQA}, \textsc{SQuALITY},
\textsc{Qasper-Abstract}, \textsc{GovReport}, \textsc{SumSCROLLS}, and
\textsc{ContractNLI}.
Baselines include Longformer-Base (4k tokens), BigBird-RoBERTa, LED, and PRIMERA.
All models are fine-tuned for 3 epochs on 4$\times$A100 GPUs with a batch size of 16.

\section{Conclusion}
\label{sec:conclusion}

We presented \modelname{}, a modular inter-layer context aggregation mechanism
for Transformers. By learning to write salient token representations to a compact
context bank and broadcasting them back via a lightweight read attention,
\modelname{} achieves larger effective attention spans at lower computational
cost. Our method is architecture-agnostic and plug-in compatible with existing
sparse-attention models. Future work will explore adaptive bank sizing and
extension to decoder-only architectures.

\paragraph{Limitations.}
The top-$k$ selection introduces a non-differentiable step; we use a straight-through
estimator during training, which may cause instability on very short sequences
($n < k$). We recommend setting $k \leq n/4$.

\paragraph{Broader Impact.}
More efficient long-context modelling reduces compute requirements for scientific
NLP pipelines, lowering barriers for researchers without access to large GPU clusters.
No direct negative societal impact is foreseen.
