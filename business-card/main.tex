\noindent\hfil\scalebox{0.9}{%
\begin{tikzpicture}[x=1pt,y=1pt]
  \pgfmathsetmacro{\cardW}{252.945}
  \pgfmathsetmacro{\cardH}{144.54}
  \pgfmathsetmacro{\cardG}{7.227}
  \pgfmathsetmacro{\totH}{5*\cardH + 4*\cardG}

  % ---- 10 cards ----
  \foreach \col in {0,1}{
    \foreach \row in {0,1,2,3,4}{
      \pgfmathsetmacro{\xp}{\col*(\cardW+\cardG)}
      \pgfmathsetmacro{\yp}{\totH - \row*(\cardH+\cardG) - \cardH}
      \begin{scope}[xshift=\xp pt, yshift=\yp pt]
        \node[anchor=south west] at (0,0) {\BusinessCard};
      \end{scope}
    }
  }

  % ---- cut-guide crosshairs at 3x6 = 18 grid intersections ----
  \pgfmathsetmacro{\gh}{8}% half tick length pt

  % x-positions of vertical cut lines
  \pgfmathsetmacro{\gxA}{0}
  \pgfmathsetmacro{\gxB}{\cardW + \cardG/2}
  \pgfmathsetmacro{\gxC}{2*\cardW + \cardG}

  % y-positions of horizontal cut lines (top to bottom)
  \pgfmathsetmacro{\gyI}{\totH}
  \pgfmathsetmacro{\gyII}{\totH - \cardH - \cardG/2}
  \pgfmathsetmacro{\gyIII}{\totH - 2*\cardH - 1.5*\cardG}
  \pgfmathsetmacro{\gyIV}{\totH - 3*\cardH - 2.5*\cardG}
  \pgfmathsetmacro{\gyV}{\totH - 4*\cardH - 3.5*\cardG}
  \pgfmathsetmacro{\gyVI}{0}

  \foreach \gx in {\gxA,\gxB,\gxC}{
    \foreach \gy in {\gyI,\gyII,\gyIII,\gyIV,\gyV,\gyVI}{
      \draw[rulecolor,line width=0.25pt,dashed]
        (\gx-\gh,\gy) -- (\gx+\gh,\gy);
      \draw[rulecolor,line width=0.25pt,dashed]
        (\gx,\gy-\gh) -- (\gx,\gy+\gh);
    }
  }

\end{tikzpicture}%
