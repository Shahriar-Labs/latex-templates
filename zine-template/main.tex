    \node[draw=#1!70, line width=0.6pt, inner xsep=5pt, inner ysep=2pt,

  \def\margin{10mm}
  \def\gap{4mm}
  \def\pw{97mm}
  \def\ph{136.5mm}
  \def\pad{3mm}

  % ── Panel background nodes (with drop shadows) ──────
  \foreach \r in {0,1} {
    \foreach \c in {0,1,2,3} {
      \pgfmathsetmacro{\px}{\margin + \c*(\pw+\gap)}
      \pgfmathsetmacro{\py}{-(\margin + \r*(\ph+\gap))}
      \node[fill=panelbg, minimum width=\pw, minimum height=\ph,
            inner sep=0pt, anchor=north west,
            drop shadow={shadow xshift=0.8pt, shadow yshift=-1.2pt, opacity=0.10}]
        at ([xshift=\px, yshift=\py] current page.north west) {};
      \draw[rust!25, line width=0.35pt]
        ([xshift=\px, yshift=\py] current page.north west)
        rectangle ++(\pw, -\ph);
    }
  }

  % ── Cut / Fold Guides ──────────────────────────────

  % Horizontal fold line (full width, dotted)
  \draw[ink!25, line width=0.35pt, densely dotted]
    ([xshift=\margin, yshift=-\margin-\ph-\gap/2] current page.north west)
    -- ++({4*\pw+3*\gap}, 0);

  % Vertical fold lines (between columns 1-2, 2-3, 3-4)
  \foreach \i in {1,2,3} {
    \pgfmathsetmacro{\vx}{\margin + \i*\pw + (\i-1)*\gap + \gap/2}
    \draw[ink!18, line width=0.3pt, loosely dotted]
      ([xshift=\vx, yshift=-\margin] current page.north west)
      -- ++(0, {-2*\ph - \gap});
  }

  % Cut zone: dashed line in the horizontal fold gap, between cols 1-2 and 2-3
  \pgfmathsetmacro{\cutxstart}{\margin + 1*\pw + 1*\gap}
  \pgfmathsetmacro{\cutxend}{\margin + 2*\pw + 2*\gap}
  \pgfmathsetmacro{\cuty}{-\margin - \ph - \gap/2}
  \draw[rust!55, line width=0.7pt, dash pattern=on 3.5pt off 2.5pt]
    ([xshift=\cutxstart, yshift=\cuty] current page.north west)
    -- ([xshift=\cutxend, yshift=\cuty] current page.north west);

  % Scissors icon at center of cut
  \pgfmathsetmacro{\cutmid}{(\cutxstart + \cutxend)/2}
  \node[rust!70, font=\tiny] at
    ([xshift=\cutmid, yshift=\cuty] current page.north west) {\faCut};

  % Fold labels
  \node[font=\fontseries{l}\selectfont\tiny\color{ink!40}, anchor=east] at
    ([xshift=\margin-2mm, yshift=-\margin-\ph-\gap/2] current page.north west) {\textit{fold}};
  \node[font=\fontseries{l}\selectfont\tiny\color{ink!40}, anchor=west] at
    ([xshift=\margin+4*\pw+3*\gap+2mm, yshift=-\margin-\ph-\gap/2] current page.north west) {\textit{fold}};

    {\RaggedRight\small\color{ink}\setlength{\baselineskip}{13pt}
      The radio hums between stations\newline
      a frequency no one broadcasts on.\newline
      I keep it there --- that gray noise ---\newline
      because silence feels too final.\strut\par}

    \vspace{2mm}
    {\RaggedRight\small\color{ink}\setlength{\baselineskip}{13pt}

      still traveling outward from 1984.\strut\par}

    \vspace{2mm}
    {\RaggedRight\small\color{ink}\setlength{\baselineskip}{13pt}
      I turn the dial with my eyes closed.\newline
      Somewhere between 98.7 and nothing\newline
      I can almost hear her humming\newline
      along to a song I've never known.\strut\par}

    \vspace{3mm}
    {\color{sage}\footnotesize\itshape --- Samir Khalil}

    \zineheading{On Repair}

    \zinebody{%
      There is a particular kind of attention required to fix something
      broken. It is not the attention of creation --- that wild, generative
      rush. It is the attention of listening. You must ask the object what
      it needs, and then you must wait for an answer.%
    }

    \vspace{2mm}
    \zinebody{%
      My grandfather repaired clocks in a shop no wider than a hallway.
      He said every broken thing tells you exactly what's wrong --- you just
      have to learn its language. A slow tick means one thing. Silence
      means another. A clock that runs fast is afraid of something.%
    }

    \vspace{2mm}
    \zinepull{We fix things not to make them new, but to make them ours.}

    \vspace{1mm}
    {\color{slate}\footnotesize\itshape --- Elena Vasquez}
