% 4. Context & Motivation
\begin{frame}{Introduction: The Need for Fast CFD}
  Computational Fluid Dynamics (CFD) is critical across science and engineering domains:
  \vspace{0.3cm}
  \begin{itemize}
    \item \textbf{Aerodynamic Optimization:} Iterative vehicle design and drag minimization.
    \item \textbf{Climate Simulation:} High-fidelity modeling of local atmospheric currents.
    \item \textbf{Medical Engineering:} Real-time blood flow tracking in cardiovascular surgeries.
  \end{itemize}
  \vspace{0.4cm}
  \begin{defenseblock}{The Scaling Bottleneck}
    Traditional solvers (FDM, FEM, FVM) require massive grid resolutions to achieve stability and accuracy, prohibiting real-time optimization or interactive loops.
  \end{defenseblock}
\end{frame}

% 5. Literature Gap (using the required baseline 'l' column table format)
\begin{frame}{Limitations of Existing Approaches}
  Prior works in fast fluid emulation fail to resolve the trade-offs between speed, accuracy, and physical compliance.
  \vspace{0.4cm}
  
  \begin{tabularx}{\linewidth}{@{}>{\color{accent}\bfseries}l@{\hspace{8mm}}>{\raggedright\arraybackslash}X@{}}
    \toprule
    Approach & Key Limitations \& Failure Modes \\
    \midrule
    Classical FVM & Computationally prohibitive for interactive applications ($O(N^3)$ scaling). \\
    Deep Emulators & Purely data-driven; generalizes poorly outside the training distribution. \\
    Physics-Informed ML & Slow convergence during training; requires pre-computed boundary fields. \\
    \bottomrule
  \end{tabularx}
  \vspace{0.4cm}
  \begin{itemize}
    \item \textbf{Our Solution:} A differentiable network that strictly embeds fluid conservation principles directly into the gradient paths.
  \end{itemize}
\end{frame}

% 11. Mathematics / Algorithm
\begin{frame}{Differentiable Navier-Stokes Loss}
  To guide convergence, we introduce a structured conservation loss:
  \begin{equation}
    \mathcal{L}_{\text{total}} = \lambda_d \mathcal{L}_{\text{data}} + \lambda_p \mathcal{L}_{\text{physics}} + \lambda_{reg} \mathcal{R}(\theta)
  \end{equation}
  
  Where the Navier-Stokes momentum residual is formulated as:
  \begin{equation}
    \mathcal{L}_{\text{physics}} = \frac{1}{N} \sum_{i=1}^N \left\| \frac{\partial \mathbf{u}_i}{\partial t} + (\mathbf{u}_i \cdot \nabla)\mathbf{u}_i + \frac{1}{\rho}\nabla p_i - \nu \nabla^2 \mathbf{u}_i \right\|^2
  \end{equation}
  \vspace{0.2cm}
  \begin{itemize}
    \item \textbf{Differentiability:} All spatial derivatives ($\nabla, \nabla^2$) are computed via automatic differentiation.
    \item \textbf{Parameter Learning:} Viscosity $\nu$ is self-calibrated during training.
  \end{itemize}
\end{frame}

% 21. Backup 1: Proof Details
\begin{frame}[noframenumbering]{Mathematical Proof: Divergence Projection}
  To ensure divergence-free flow, we solve the Poisson equation for pressure:
  \begin{equation}
    \nabla^2 p = \rho \nabla \cdot \left( \mathbf{u}^* \right)
  \end{equation}
  \vspace{0.2cm}
  where $\mathbf{u}^*$ is the intermediate velocity field. The correction step is:
  \begin{equation}
    \mathbf{u} = \mathbf{u}^* - \frac{1}{\rho}\nabla p
  \end{equation}
  
  \begin{defenseblock}{Guaranteed Divergence-Free Flow}
    Taking the divergence of the correction step:
    \begin{equation}
      \nabla \cdot \mathbf{u} = \nabla \cdot \mathbf{u}^* - \frac{1}{\rho}\nabla^2 p = 0
    \end{equation}
    This projection step is implemented as a fixed differentiable layer in our network.
  \end{defenseblock}
\end{frame}
