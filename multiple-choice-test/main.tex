\begin{mcoptions}

  \item[\textbf{B.}] \textbf{Stack} --- the most recently added element is removed first.
  \item[\textbf{C.}] \textbf{Deque} --- elements can be added or removed from either end.
  \item[\textbf{D.}] \textbf{Priority Queue} --- elements are removed based on assigned priority.
\end{mcoptions}
