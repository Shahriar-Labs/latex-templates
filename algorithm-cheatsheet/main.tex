% --- CORE DATA STRUCTURES ---
\begin{topicbox}{\faCode~Core Data Structures Notes}
  \scriptsize
  \begin{description}[leftmargin=3pt,itemsep=2pt,topsep=2pt,parsep=0pt]
    \item[\color{primary}Singly-Linked List:] Nodes store value and \texttt{next} pointer. Efficient insertion/deletion at head ($\mathcal{O}(1)$), but linear lookup.
    \item[\color{primary}Doubly-Linked List:] Nodes store \texttt{next} and \texttt{prev} pointers. Allows bidirectional traversal; $\mathcal{O}(1)$ deletion when node reference is given.
    \item[\color{primary}Hash Table:] Key-value store using a hash function. Collisions resolved via Chaining (linked lists) or Open Addressing (probing).
    \item[\color{primary}Binary Search Tree (BST):] Left child $<$ parent $\le$ right child. Inorder traversal yields sorted elements.
    \item[\color{primary}AVL Tree:] Self-balancing BST where heights of two child subtrees differ by at most one. Uses rotations to rebalance.
  \end{description}
\end{topicbox}

% --- QUICKSORT IMPLEMENTATION ---
\begin{topicbox}{\faCode~Quicksort (Lomuto Partition)}
  \scriptsize Partition-based divide-and-conquer algorithm. Average-case optimal, but degrades to $\mathcal{O}(n^2)$ on sorted arrays with poor pivot choices.
  \vspace{3pt}
  \begin{lstlisting}
def quicksort(arr, low, high):
    if low < high:
        p = partition(arr, low, high)
        quicksort(arr, low, p - 1)
        quicksort(arr, p + 1, high)

def partition(arr, low, high):
    pivot = arr[high]
    i = low - 1
    for j in range(low, high):
        if arr[j] <= pivot:
            i += 1
            arr[i], arr[j] = arr[j], arr[i]
    arr[i+1], arr[high] = arr[high], arr[i+1]
    return i + 1
  \end{lstlisting}
\end{topicbox}

% --- SEARCHING ALGORITHMS ---
\begin{accentbox}{\faSearch~Searching Algorithms}
  \scriptsize
  \begin{description}[leftmargin=3pt,itemsep=2pt,topsep=2pt,parsep=0pt]
    \item[\color{primary}Linear Search:] Check each element. $\mathcal{O}(n)$ time.
    \item[\color{primary}Binary Search:] Half-interval search on sorted arrays. $\mathcal{O}(\log n)$ time, $\mathcal{O}(1)$ space.
  \end{description}
  \vspace{2pt}
  \begin{lstlisting}
def binary_search(arr, target):
    low, high = 0, len(arr) - 1
    while low <= high:
        mid = (low + high) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            low = mid + 1
        else:
            high = mid - 1
    return -1
  \end{lstlisting}
\end{accentbox}

% --- GRAPH TRAVERSAL: BFS ---
\begin{topicbox}{\faProjectDiagram~Breadth-First Search (BFS)}
  \scriptsize Traverses level-by-level using a FIFO queue. Finds shortest path in unweighted graphs. Time: $\mathcal{O}(V + E)$, Space: $\mathcal{O}(V)$.
  \vspace{3pt}
  \begin{lstlisting}
from collections import deque

def bfs(graph, start):
    visited = {start}
    queue = deque([start])
    while queue:
        node = queue.popleft()
        for neighbor in graph[node]:
            if neighbor not in visited:
                visited.add(neighbor)
                queue.append(neighbor)
  \end{lstlisting}
\end{topicbox}

% --- GRAPH TRAVERSAL: DFS ---
\begin{topicbox}{\faProjectDiagram~Depth-First Search (DFS)}
  \scriptsize Traverses deep along paths using a recursion stack. Used for topological sorting and cycle detection. Time: $\mathcal{O}(V + E)$, Space: $\mathcal{O}(V)$.
  \vspace{3pt}
  \begin{lstlisting}
def dfs(graph, node, visited=None):
    if visited is None:
        visited = set()
    visited.add(node)
    for neighbor in graph[node]:
        if neighbor not in visited:
            dfs(graph, neighbor, visited)
    return visited
  \end{lstlisting}
\end{topicbox}

% --- HEAPS & STACKS ---
\begin{topicbox}{\faLayerGroup~Stack, Queue \& Heaps}
  \scriptsize
  \begin{description}[leftmargin=3pt,itemsep=2pt,topsep=2pt,parsep=0pt]
    \item[\color{primary}Stack:] LIFO (Last-In-First-Out). Operations: Push/Pop are $\mathcal{O}(1)$. Used in function calls, backtracking.
    \item[\color{primary}Queue:] FIFO (First-In-First-Out). Operations: Enqueue/Dequeue are $\mathcal{O}(1)$. Used in BFS, scheduling.
    \item[\color{primary}Min-Heap / Max-Heap:] Complete binary tree where parent is always smaller/larger than children. Insertion/Extraction are $\mathcal{O}(\log n)$, Peek is $\mathcal{O}(1)$.
  \end{description}
\end{topicbox}

\begin{accentbox}{\faNetworkWired~Advanced Graph Algorithms}
  \scriptsize
  \begin{description}[leftmargin=3pt,itemsep=2pt,topsep=2pt,parsep=0pt]
    \item[\color{primary}Dijkstra's Algorithm:] Finds shortest path from single source in non-negative weighted graph. Uses Min-Priority Queue. Time: $\mathcal{O}((V + E)\log V)$.

% --- DYNAMIC PROGRAMMING ---
\begin{topicbox}{\faBrain~Dynamic Programming Principles}
  \scriptsize
  Solves complex problems by breaking them into overlapping subproblems, storing results (memoization/tabulation) to avoid recomputation.
  \vspace{2pt}
  
  \textbf{0/1 Knapsack recurrence relation:}
  \[ DP[i][w] = \max(DP[i-1][w], DP[i-1][w-w_i] + v_i) \]
  
  \begin{lstlisting}
# 0/1 Knapsack - Space Optimized Tabulation
def knapsack(W, weights, values, n):
    dp = [0] * (W + 1)
    for i in range(n):
        for w in range(W, weights[i] - 1, -1):
            dp[w] = max(dp[w], dp[w - weights[i]] + values[i])
    return dp[W]
  \end{lstlisting}
\end{topicbox}

% --- TREE ROTATIONS (AVL) ---
\begin{topicbox}{\faTree~AVL Tree Rotations}
  \scriptsize
  AVL trees maintain a balance factor $|h_L - h_R| \le 1$. Restores balance using local transformations (rotations) in $\mathcal{O}(1)$ time.
  \vspace{2pt}
  \begin{lstlisting}
def right_rotate(y):
    x = y.left
    T2 = x.right
    # Perform rotation
    x.right = y
    y.left = T2
    # Update heights
    y.height = 1 + max(height(y.left), height(y.right))
    x.height = 1 + max(height(x.left), height(x.right))
    return x
  \end{lstlisting}
  \vspace{2pt}
  \textbf{Four Rotation Cases:}
  \begin{description}[leftmargin=3pt,itemsep=1pt,topsep=1pt,parsep=0pt]
    \item[\color{primary}Left-Left (LL):] Single Right Rotation.
    \item[\color{primary}Right-Right (RR):] Single Left Rotation.
    \item[\color{primary}Left-Right (LR):] Left Rotate child, then Right Rotate parent.
    \item[\color{primary}Right-Left (RL):] Right Rotate child, then Left Rotate parent.
  \end{description}
\end{topicbox}
