% ── 1. SAFETY INFORMATION ────────────────────────────────────
\section{Safety Information}

\begin{dangerbox}
  \textbf{Electric shock hazard.} Do not expose the charging case to liquids or insert foreign
  objects into the USB-C port. Never attempt to open or repair the earbuds yourself —
  there are no user-serviceable parts inside.
\end{dangerbox}

\vspace{4pt}

\begin{warningbox}
  \textbf{Hearing protection.} Prolonged exposure to volumes above 85\,dB can cause permanent
  hearing damage. Begin at low volume and increase gradually. Follow the \textbf{60/60 rule}:
  maximum 60\% volume for no more than 60 minutes at a time.
\end{warningbox}

\vspace{4pt}

\begin{warningbox}[title={\faCar\enspace\textbf{Traffic Safety}}, colframe=warnborder, coltitle=warnborder]
  Do not use Active Noise Cancellation (ANC) in traffic or other environments where
  situational awareness is essential. Use \textbf{Transparency Mode} instead.
\end{warningbox}

\vspace{4pt}

\begin{itemize}[itemsep=2pt, parsep=0pt, leftmargin=1.4em]
  \item[\faCheckCircle] Keep out of reach of children under 3 years (small parts).
  \item[\faCheckCircle] Operating temperature: 0\,\textdegree{}C to 40\,\textdegree{}C (32\,\textdegree{}F to 104\,\textdegree{}F).
  \item[\faCheckCircle] Store in the provided charging case when not in use.
  \item[\faCheckCircle] Clean with a dry, lint-free cloth only — do not use solvents.
\end{itemize}

\subsection{Charging for the First Time}

Place the earbuds in the charging case and connect via USB-C. A full charge from empty takes

on the front of the case shows charge status (see Section~\ref{sec:features}).

% ── 3. SETUP & INSTALLATION ──────────────────────────────────
\section{Setup \& Installation}

\begin{steps}
  \item \textbf{Charge the earbuds.} Place both earbuds in the case and connect USB-C until
    the case LED shows solid white (fully charged).
  \item \textbf{Open the case lid} near your smartphone. The earbuds enter pairing mode
    automatically — the LED pulses blue/white.
  \item \textbf{On your device}, open \textit{Settings} $\to$ \textit{Bluetooth} and select
    \textbf{``SoundWave Pro X1''} from the list. Pairing completes within 5 seconds.
  \item \textbf{iOS / Android App (optional).} Download the \textbf{SoundWave Connect} app
    from the App Store or Nexa Play for EQ presets, firmware updates, and find-my-earbuds.
  \item \textbf{Fit the ear tips.} Twist the earbud into your ear canal with the stem angled
    downward. Run the \textit{Ear Tip Fit Test} in the app for a proper acoustic seal.
  \item \textbf{Multi-device pairing.} To pair a second device, open Bluetooth settings on
    that device and select \textbf{``SoundWave Pro X1''}. The earbuds remember up to
    \textbf{8 paired devices} and auto-connect to the 2 most recent.
\end{steps}

\begin{tipbox}
  \textbf{Swift Switch.} To switch audio from one connected device to another, simply play
  audio on the target device — the earbuds switch automatically in under 1 second.
\end{tipbox}

\subsection{LED Indicator Guide}

\begin{tabularx}{\linewidth}{@{}K@{\hspace{8mm}}>{\raggedright\arraybackslash}X@{}}
\setlength{\parskip}{0pt}%
Solid white       & Fully charged \\
Pulsing white     & Charging in progress \\
Pulsing blue/white & Pairing mode \\
Solid blue (3s)   & Connected successfully \\
Pulsing red       & Battery low ($<$15\%) \\
Solid red         & Battery critical — charge immediately \\
\end{tabularx}

\subsection{Active Noise Cancellation \& Transparency}

The Pro X1 uses dual-feed-forward microphones with a \textbf{$-38$\,dB} adaptive ANC
algorithm. Transparency Mode passes through ambient sound at near-natural levels,
ideal for conversations without removing the earbuds.

\begin{tipbox}
  ANC performance is maximised with a proper ear tip seal. If you notice reduced
  cancellation, try the next larger ear-tip size or run the Fit Test in the app.
\end{tipbox}

% ── 5. TROUBLESHOOTING ───────────────────────────────────────
\section{Troubleshooting}

\renewcommand{\arraystretch}{1.3}
\setlength{\parskip}{0pt}

\begin{tabularx}{\linewidth}{@{}
  >{\color{accent}\bfseries\raggedright}p{52mm}
  @{\hspace{6mm}}
  >{\raggedright\arraybackslash}X
@{}}
\toprule
\textcolor{accent}{Problem} & \textbf{Solution} \\
\midrule
Earbuds not found in Bluetooth list
  & Ensure case lid is open and earbuds are removed. If still not visible, hold both
    earbuds' stems for 6 seconds until LED flashes red/blue (factory reset), then retry. \\
\addlinespace
Audio drops or cuts out
  & Move device within 10\,m and remove obstructions. Disable Wi-Fi 2.4\,GHz on device
    temporarily to check for interference. \\
\addlinespace
One earbud has lower volume
  & Clean the speaker mesh with a dry brush. Check that balance is centred in device
    accessibility settings. \\
\addlinespace
Case not charging
  & Ensure USB-C cable is fully seated. Try a different USB port or charger ($\geq$5\,W).
    Check for debris in the port. \\
\addlinespace
Poor microphone quality on calls
  & Ensure ANC is enabled (mic uses feedforward path). Keep stems facing downward. \\
\addlinespace
Earbuds won't turn on
  & Place in case for at least 10 minutes, then remove and re-insert. If LED remains off,
    perform a factory reset (see above). \\
\addlinespace
ANC seems ineffective
  & Run the Ear Tip Fit Test in the SoundWave Connect app. An improper seal reduces
    ANC effectiveness by up to 60\%. \\
\bottomrule
\end{tabularx}
