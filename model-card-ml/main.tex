\section{Model Details}

\begin{infobox}
  \textbf{Summary.} Lumina-7B-Instruct is a 7.3-billion-parameter decoder-only
  Transformer language model, fine-tuned via supervised instruction tuning and
  RLHF for general-purpose conversational and coding tasks. It supports a context
  window of 32,768 tokens and is released under the TensorForge Community License.
\end{infobox}

\vspace{6pt}

\begin{tabularx}{\linewidth}{@{}>{\color{accent}\bfseries\small}l@{\hspace{7mm}}>{\raggedright\arraybackslash\small}X@{}}
  \detailrow{Organization}{TensorForge AI (tensorforge.ai)}
  \detailrow{Model Type}{Causal Language Model — decoder-only Transformer}
  \detailrow{Version}{2.1.0 (3 July 2026)}
  \detailrow{License}{TensorForge Community License v1.0}
  \detailrow{Parameters}{$7.3\times 10^{9}$ (7.3B)}
  \detailrow{Architecture}{32 layers, 4096 hidden size, 32 attention heads, Grouped-Query Attention (4 KV heads), RoPE embeddings, SwiGLU activations}
  \detailrow{Tokenizer}{SentencePiece BPE, vocabulary size 128\,000}
  \detailrow{Context Length}{32\,768 tokens (input + output)}
  \detailrow{Training Data}{Mix of filtered web text (CommonCrawl), books, Wikipedia, GitHub code, and curated instruction datasets. Knowledge cutoff: December 2025.}
  \detailrow{Input / Output}{Text in, text out. Supports system prompts and multi-turn conversation.}
  \detailrow{Languages}{English (primary); 30+ languages at varying proficiency}
  \detailrow{Hardware}{256 $\times$ NVIDIA H100-80GB GPUs; 18 days training}
  \detailrow{Precision}{BF16 training; FP16 inference}
  \detailrow{Fine-Tuning}{Supervised fine-tuning + RLHF (PPO) on 180K human preference pairs}
  \detailrow{Contact}{\href{mailto:safety@tensorforge.ai}{safety@tensorforge.ai}~~~\textbullet~~~\href{https://tensorforge.ai/models/lumina}{tensorforge.ai/models/lumina}}
\end{tabularx}

\section{Example Usage}

\begin{lstlisting}[language=Python, caption={Inference with the Transformers library.}]
from transformers import AutoModelForCausalLM, AutoTokenizer

model_id = "tensorforge/Lumina-7B-Instruct"
tokenizer = AutoTokenizer.from_pretrained(model_id)
model = AutoModelForCausalLM.from_pretrained(
    model_id,
    torch_dtype="auto",
    device_map="auto",
)

messages = [
    {"role": "system", "content": "You are a helpful assistant."},
    {"role": "user",   "content": "Explain backpropagation in 3 sentences."},
]
inputs = tokenizer.apply_chat_template(
    messages, tokenize=True, return_tensors="pt"
).to(model.device)

outputs = model.generate(inputs, max_new_tokens=256, temperature=0.7)
print(tokenizer.decode(outputs[0], skip_special_tokens=True))
\end{lstlisting}

\section{Caveats \& Limitations}
\label{sec:caveats}

\begin{caveatbox}
  \textbf{\textcolor{warn}{\faExclamationTriangle~~Critical Caveats}}

  \vspace{2pt}
  \begin{itemize}
    \item \textbf{Hallucination.} Like all LLMs, Lumina-7B-Instruct may
    generate plausible-sounding but factually incorrect content. Always verify
    outputs in high-stakes settings.
    \item \textbf{Bias \& Stereotypes.} Training data reflects web-scale biases.
    The model may reproduce stereotypes, particularly for under-represented
    groups (see Fairness Analysis, Section~4).
    \item \textbf{Out-of-Scope Use.} This model has \emph{not} been validated
    for medical diagnosis, legal advice, safety-critical systems, or decisions
    with material impact on individuals' rights.
    \item \textbf{Adversarial Robustness.} The model is vulnerable to prompt
    injection, jailbreaking, and extraction attacks. Deploy with input/output
    filtering in production.
    \item \textbf{Privacy.} Training data was filtered for PII, but residual
    memorisation of rare sequences is possible. Do not expose the model to
    sensitive user data unless additional privacy-preserving measures are in
    place.
    \item \textbf{Multilingual Fairness.} Performance degrades significantly for
    low-resource languages. Users should evaluate independently for their target
    languages.
  \end{itemize}
\end{caveatbox}

% ── Recommendations ──
\subsection*{Recommendations for Downstream Use}

\begin{itemize}
  \item Perform \textbf{domain-specific evaluation} on your target task and
  demographic distribution before deployment.
  \item Apply \textbf{output guardrails} (toxicity classifiers, factuality
  checkers) in user-facing applications.
  \item Monitor \textbf{drift and feedback loops} in production—model behaviour
  can shift with changing usage patterns.
  \item Where feasible, provide users with \textbf{transparency mechanisms}:
  confidence scores, source attribution, or an opt-out for AI-generated content.
\end{itemize}
