% Custom photo placeholder macro
\newcommand{\photoplaceholder}[3]{%
  % #1: width in cm
  % #2: height in cm
  % #3: label
  \begin{tikzpicture}
    \pgfmathsetmacro{\w}{#1}
    \pgfmathsetmacro{\h}{#2}
    \pgfmathsetmacro{\midx}{\w/2}
    \pgfmathsetmacro{\midy}{\h/2}
    \pgfmathsetmacro{\icony}{\midy + 0.25}
    \pgfmathsetmacro{\tboxy}{\midy - 0.45}
    
    \draw[draw=border-color, fill=white, line width=1.2pt] (0,0) rectangle (\w,\h);
    \draw[draw=border-color!50, dashed, line width=0.8pt] (0.08,0.08) rectangle (\w-0.08,\h-0.08);
    \node[text=border-color!85] at (\midx,\icony) {\fontsize{18pt}{18pt}\selectfont \faCamera};
    \node[text=text-muted, font=\fontsize{7.5pt}{9pt}\selectfont\bfseries] at (\midx,\tboxy) {\MakeUppercase{#3}};
  \end{tikzpicture}
}
