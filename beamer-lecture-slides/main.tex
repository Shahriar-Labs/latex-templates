\section{The Consensus Problem}

\begin{frame}{The Consensus Problem}
\begin{tcolorbox}[defblock,title={Definition (Consensus)}]
A set of $n$ distributed processes must agree on a single output value from a set of proposed input values, even in the presence of failures.
\end{tcolorbox}

\pause

\vspace{6pt}
\textbf{Required Properties:}
\begin{itemize}
  \item<2-> \textbf{Agreement:} Every correct process outputs the same value.
  \item<3-> \textbf{Termination:} Every correct process eventually decides.
  \item<4-> \textbf{Validity:} The output value was proposed by some process.
  \item<5-> \textbf{Integrity:} Each process decides at most once.
\end{itemize}
\end{frame}

\begin{frame}{Why Consensus Matters}
\textbf{Real-world applications:}

\vspace{8pt}
\begin{itemize}
  \item \textbf{Leader Election:} Selecting a coordinator among a group of servers. Used in NimbusDB for automatic failover across data centers.

  \vspace{6pt}
  \item \textbf{State Machine Replication:} Keeping multiple replicas consistent. The Apex Cloud platform uses replicated state machines for its configuration service.

  \vspace{6pt}
  \item \textbf{Atomic Broadcast:} Ensuring all nodes receive messages in the same order. Vertex's trading engine depends on total-order broadcast for audit trails.
\end{itemize}

\vspace{10pt}
\begin{tcolorbox}[exblock,title={Key Insight}]
Consensus is the foundational primitive that enables fault-tolerant distributed coordination.
\end{tcolorbox}
\end{frame}

\section{FLP Impossibility}

\begin{frame}{FLP Impossibility Result}
\begin{tcolorbox}[thmblock,title={Theorem (Fischer--Lynch--Paterson, 1985)}]
In a fully asynchronous distributed system, it is impossible to achieve deterministic consensus in the presence of even a single crash failure.
\end{tcolorbox}

\pause

\vspace{10pt}
\textbf{What this means:}
\begin{itemize}
  \item<2-> No deterministic protocol can guarantee both \textbf{safety} (agreement) and \textbf{liveness} (termination) under full asynchrony.
  \item<3-> The adversary can delay messages arbitrarily, preventing a process from distinguishing a slow process from a crashed one.
  \item<4-> This is a \textbf{fundamental result} --- not an engineering limitation but a theoretical impossibility.
\end{itemize}

\pause

\vspace{10pt}
\textbf{Implication:} Practical systems must either assume partial synchrony or use randomization to circumvent FLP.
\end{frame}

\section{The Paxos Protocol}

\begin{frame}{Paxos: Core Ideas}
\textbf{Developed by Leslie Lamport (1989).} Named after the legislative process on the Greek island of Paxos.

\vspace{10pt}
\textbf{Three Roles:}
\begin{itemize}
  \item \textbf{Proposers:} Initiate consensus by proposing values.
  \item \textbf{Acceptors:} Vote on proposals; a \emph{majority quorum} must agree.
  \item \textbf{Learners:} Learn the chosen value once consensus is reached.
\end{itemize}

\vspace{8pt}
\textbf{Key Insight:} A process can play multiple roles simultaneously.

\vspace{8pt}
\begin{tcolorbox}[defblock,title={Safety Property}]
If any value is chosen, every subsequent chosen value must be identical. Paxos guarantees this even with $f < n/2$ failures.
\end{tcolorbox}
\end{frame}

\begin{frame}{Paxos: Phase 1 --- Prepare \& Promise}
\textbf{Goal:} Elect a leader and learn about any previously chosen values.

\vspace{8pt}
\begin{enumerate}
  \item<2-> \textbf{Proposer} selects a unique, monotonically increasing proposal number $n$.
  \item<3-> Sends $\textsc{Prepare}(n)$ to a \textbf{majority} of acceptors.
  \item<4-> Each \textbf{Acceptor} that receives $\textsc{Prepare}(n)$:
    \begin{itemize}
      \item If $n$ is the highest it has seen, it \textbf{promises} to reject proposals with lower numbers.
      \item It replies with $\textsc{Promise}(n, v, n')$ where $v$ is the highest-numbered proposal it has already accepted (with number $n' < n$), or $\bot$ if none.
    \end{itemize}
  \item<5-> The proposer collects promises. If it hears from a majority, Phase~2 begins.
\end{enumerate}

\vspace{6pt}
\onslide<6->{
\begin{tcolorbox}[exblock,title={Invariant}]
An acceptor always remembers the highest proposal number it has promised and the highest-numbered proposal it has accepted.
\end{tcolorbox}
}
\end{frame}

\begin{frame}{Paxos: Phase 2 --- Accept \& Learn}
\textbf{Goal:} Drive the system to choose a single value.

\vspace{8pt}
\begin{enumerate}
  \item<2-> The \textbf{Proposer} examines the Promise replies:
    \begin{itemize}
      \item If any acceptor returned a previously accepted value, the proposer \textbf{must} use the value with the highest $n'$.
      \item Otherwise, it is free to propose its own value $v$.
    \end{itemize}
  \item<3-> Proposer sends $\textsc{Accept}(n, v)$ to the same majority.
  \item<4-> Each \textbf{Acceptor} accepts $(n, v)$ \emph{only if} it has not promised a proposal $> n$.
  \item<5-> When a majority of acceptors have accepted $(n, v)$, the value $v$ is \textbf{chosen}.
  \item<6-> \textbf{Learners} are notified of the chosen value.
\end{enumerate}

\vspace{6pt}
\onslide<7->{
\begin{tcolorbox}[defblock,title={Chosen Value}]
A value is \emph{chosen} once a majority of acceptors have accepted it for the same proposal number. All future decisions will agree on this value.
\end{tcolorbox}
}
\end{frame}

\section{Key Takeaways}

\begin{frame}{Summary \& Key Takeaways}
\begin{enumerate}
  \item \textbf{Consensus} is the problem of getting distributed processes to agree on a single value despite failures.

  \vspace{4pt}
  \item The \textbf{FLP impossibility} result proves that deterministic consensus is impossible in a purely asynchronous model with even one crash.

  \vspace{4pt}
  \item \textbf{Paxos} provides a practical solution by assuming eventual synchrony (or using randomization), achieving safety under all conditions.

  \vspace{4pt}
  \item The two-phase structure --- \textbf{Prepare/Promise} followed by \textbf{Accept/Learn} --- is the core mechanism.

  \vspace{4pt}
  \item Modern systems (NimbusDB, Apex Cloud, Vertex) build on these primitives for fault-tolerant coordination.
\end{enumerate}
