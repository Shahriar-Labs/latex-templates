% ── SCHEME 3 — mechanism with curved arrows ─────────────────────────────────
\section{Reaction mechanism}
\scheme{3}{Bimolecular nucleophilic substitution (\ce{S_N2}): hydroxide attacks
the carbon of bromomethane, displacing bromide via a backside attack.}

\vspace{6pt}
\begin{center}
\schemestart
  \chemfig{@{nu}\chemabove{O}{\scriptstyle\ominus}H}
  \+
  \chemfig{H_3C-@{c}C(-[6]H)(-[2]H)-@{lg}Br}
  \arrow{->}
  \chemfig{H_3C-OH}
  \+
  \chemfig{\chembelow{Br}{\scriptstyle\ominus}}
\schemestop
\chemmove{
  \draw[->,shorten <=2pt,shorten >=2pt,accent,line width=0.7pt]
    (nu) .. controls +(20:6mm) and +(160:6mm) .. (c);
  \draw[->,shorten <=2pt,shorten >=2pt,accent,line width=0.7pt]
    (c) .. controls +(-20:6mm) and +(200:6mm) .. (lg);
}
\end{center}

\vspace{2pt}
{\color{muted}\small The hydroxide lone pair forms the new \ce{C-O} bond as the
\ce{C-Br} bond breaks heterolytically, giving methanol and bromide in a single
concerted step with inversion of configuration.}
