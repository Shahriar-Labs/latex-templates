% Draw a stylised Khanda () in the accent colour
\newcommand{\drawkhanda}{%
  \begin{tikzpicture}[scale=0.35, line cap=round, line join=round]
    % Chakkar (circle)
    \draw[accent, line width=1.2pt] (0,0.5) circle (0.75);
    % Central double-edged sword shaft
    \draw[accent, line width=1.3pt] (0,-0.85) -- (0,2.1);
    % Sword tip
    \fill[accent] (-0.13,2.0) -- (0,2.1) -- (0.13,2.0) -- cycle;
    % Handle crossbar
    \draw[accent, line width=1.6pt] (-0.45,-0.85) -- (0.45,-0.85);
    % Handle grip below
    \draw[accent, line width=0.9pt] (0,-1.15) -- (0,-0.85);
    \draw[accent, line width=0.9pt] (-0.06,-1.2) -- (0.06,-1.2);
    % Left kirpan (curved blade)
    \draw[accent, line width=1.0pt] (-0.55,0.95)
      .. controls (-0.5,0.65) and (-0.35,0.45) .. (-0.35,0.1) -- (-0.25,0.2)
      .. controls (-0.38,0.5) and (-0.42,0.65) .. (-0.45,0.8) -- cycle;
    % Right kirpan (mirrored)
    \draw[accent, line width=1.0pt] (0.55,0.95)
      .. controls (0.5,0.65) and (0.35,0.45) .. (0.35,0.1) -- (0.25,0.2)
      .. controls (0.38,0.5) and (0.42,0.65) .. (0.45,0.8) -- cycle;
  \end{tikzpicture}%
}
