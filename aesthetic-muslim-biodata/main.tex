% Rub el Hizb (8-pointed Islamic Star) command
\newcommand{\rubelhizb}[3]{% #1 = center, #2 = size, #3 = inner fill color
  \begin{scope}[shift={#1}]
    \fill[accentgold] (-#2,-#2) rectangle (#2,#2);
    \fill[accentgold, rotate=45] (-#2,-#2) rectangle (#2,#2);
    \fill[#3] (0,0) circle (#2*0.42);
  \end{scope}
}
