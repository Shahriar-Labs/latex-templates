% --- Custom Chord Diagram Macro ---
% #1: Chord Name (e.g. C)
% #2: Start Fret (e.g. 1)
% #3: Top markers list (6 elements: x, o, or e)
% #4: Dots list (comma-separated string/fret/finger, e.g. 2/3/3)
% #5: Barre list (comma-separated fret/string-start/string-end, e.g. 1/1/6)
\newcommand{\chord}[5]{%
  \begin{minipage}[t]{2.7cm}%
    \centering%
    \vspace{1mm}%
    {\large\bfseries\color{accent}#1}\par\vspace{2mm}%
    \begin{tikzpicture}[x=4.2mm, y=5.5mm, line width=0.6pt, color=dark]%
      \useasboundingbox (1,0.5) rectangle (6,-5.2);
      % Draw the fret grid
      \foreach \y in {0,...,-5} {
        \draw[grid, line width=0.6pt] (1,\y) -- (6,\y);
      }
      \foreach \x in {1,...,6} {
        \draw[grid, line width=0.6pt] (\x,0) -- (\x,-5);
      }
      
      % Draw thick nut if start fret is 1
      \ifthenelse{\equal{#2}{1}}{
        \draw[accent, line width=2.2pt] (1,0) -- (6,0);
      }{
        % Otherwise draw start fret label on the left
        \node[left, font=\scriptsize\sffamily\bfseries\color{muted}] at (0.8,-0.5) {#2 fr};
      }
      
      % Draw top markers (x, o)
      \foreach \m [count=\i] in {#3} {
        \ifthenelse{\equal{\m}{x}}{
          \node[font=\scriptsize\sffamily\bfseries\color{muted}] at (\i,0.3) {$\times$};
        }{}
        \ifthenelse{\equal{\m}{o}}{
          \node[font=\scriptsize\sffamily\bfseries\color{muted}] at (\i,0.3) {$\circ$};
        }{}
      }
      
      % Draw Barre(s) if provided
      \ifthenelse{\equal{#5}{}}{}{
        \foreach \f/\s/\e in {#5} {
          \pgfmathsetmacro{\ypos}{-(\f - #2 + 0.5)}
          \fill[accentlight!85, rounded corners=1.8pt] (\s-0.16,\ypos-0.16) rectangle (\e+0.16,\ypos+0.16);
        }
      }
      
      % Draw Dots with fingerings
      \ifthenelse{\equal{#4}{}}{}{
        \foreach \s/\f/\fing in {#4} {
          \pgfmathsetmacro{\ypos}{-(\f - #2 + 0.5)}
          \fill[accentlight] (\s,\ypos) circle (2.1mm);
          \node[white, font=\scriptsize\bfseries\sffamily] at (\s,\ypos) {\fing};
        }
      }
    \end{tikzpicture}%
  \end{minipage}%
}

% --- Practice Tips Section ---
\begin{tipsbox}{Practice Tips for Clean Chords}
  \begin{itemize}[leftmargin=*, noitemsep, topsep=1pt, partopsep=0pt, parsep=2pt]
    \item \textbf{Clean Execution:} Use the tips of your fingers to press the strings. Position them as close to the fret wire as possible without touching it to prevent buzzing.
    \item \textbf{Precise Muting:} Ensure strings marked with a $\times$ at the top are muted by lightly touching them with a finger or thumb, or avoid strumming them entirely.
    \item \textbf{Mastering Barres:} For barre chords (like F, Bm, and Cm), roll your index finger slightly to the side to use the harder bony edge for a clearer tone.
    \item \textbf{Daily Progression:} Practice switching between common transitions like \textbf{C -- G -- Am -- F} to build muscle memory and speed.
  \end{itemize}
\end{tipsbox}
