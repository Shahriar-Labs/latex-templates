                         .. controls (0.2,0.9) and (0.1,0.4) .. (0,0.05);
    % Left petal 1
    \path[fill=maroon] (0,0.05) .. controls (-0.4,0.3) and (-0.5,0.8) .. (-0.4,1.1) 
                         .. controls (-0.2,0.9) and (-0.1,0.4) .. (0,0.05);

                         .. controls (0.4,0.6) and (0.2,0.3) .. (0,0.1);
    % Left petal 2
    \path[fill=maroon] (0,0.1) .. controls (-0.5,0.1) and (-0.7,0.5) .. (-0.6,0.8)

                            .. controls (0.1,-0.05) and (-0.1,-0.05) .. (-0.3,-0.1);
  \end{tikzpicture}%
