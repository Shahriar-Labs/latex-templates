\begin{frame}[fragile]{The Migration Challenge}

  \begin{columns}[T,onlytextwidth]
    \column{0.52\textwidth}
    \begin{infocard}
      {\large\bfseries\color{accentAmber}\faExclamationTriangle\hspace{8pt}Why Migrate?}

      \vspace{8pt}
      The v1 client served us well for two years, but production incidents at Apex and Meridian exposed critical gaps:
      \vspace{4pt}
      \begin{itemize}
        \item Sync-only calls block the event loop under load
        \item No built-in retry or circuit breaking
        \item Token refresh requires manual intervention
        \item Response parsing couples callers to raw HTTP
      \end{itemize}
    \end{infocard}

    \vspace{10pt}
    \begin{infocard}
      {\large\bfseries\color{accentCyan}\faBullseye\hspace{8pt}Target Architecture}

      \vspace{6pt}
      \highlight{async-first} design with typed service objects, automatic token rotation, and exponential backoff. Zero breaking changes to business logic.
    \end{infocard}

    \column{0.48\textwidth}
    \begin{codeblock}{nexa\_client\_v1.py}
import requests

class NexaClient:
    def __init__(self, api_key):
        self.key = api_key
        self.base = "https://api.nexa.io"

    def fetch_data(self, endpoint):
        url = f"{self.base}{endpoint}"
        resp = requests.get(url,
            headers={"X-Key": self.key})
        if resp.status_code != 200:
            raise APIError(resp.text)
        return resp.json()

    def post_data(self, endpoint, body):
        url = f"{self.base}{endpoint}"
        resp = requests.post(url,
            json=body,
            headers={"X-Key": self.key})
        return resp.json()
\end{codeblock}
  \end{columns}

\end{frame}

    \vspace{6pt}
    \begin{infocard}
      {\small\faTimesCircle\hspace{4pt}\color{accentRed}Pain points:} No retries, raw dicts everywhere, blocking I/O, key leaks in logs.
    \end{infocard}

    \column{0.5\textwidth}
    {\small\color{accentCyan}\faArrowRight\hspace{4pt} v2 \textemdash{} fluent builder}
    \vspace{4pt}
    \begin{codeblock}{new\_approach.py}
from nexa import NexaClientV2
from nexa.auth import ApiKeyAuth
from nexa.services import UserService

client = (
    NexaClientV2.configure()
    .with_auth(ApiKeyAuth.from_env())
    .with_retry(max_attempts=3,
                backoff_factor=0.5)
    .with_timeout(30)
    .build()
)

users = client.users.list(
    limit=50, status="active"
)
# Returns List[User] not raw dict
for u in users:
    print(u.name, u.email)
\end{codeblock}

    \vspace{6pt}
    \begin{infocard}
      {\small\faCheckCircle\hspace{4pt}\color{accentCyan}Wins:} Typed models, async-ready, auto-retry, env-based secrets.
    \end{infocard}
