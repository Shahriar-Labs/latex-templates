\section{Overview}
The Nimbus API is a resource-oriented REST interface for managing files, folders, and
event webhooks inside a Nimbus workspace. All requests and responses use JSON, all
endpoints require HTTPS, and all timestamps are ISO\,8601 UTC. Path-based versioning
(\texttt{/v2}); \texttt{/v1} is deprecated and sunsets 2026-12-31.

\begin{itemize}[leftmargin=14pt,itemsep=1pt,topsep=1pt]
  \item Resource IDs (\texttt{file\_id}, \texttt{folder\_id}, \dots) are opaque, URL-safe
    strings prefixed by type, e.g. \texttt{fl\_8a2c\dots} for files.
  \item List endpoints are paginated with \texttt{cursor} and \texttt{limit}; see
    Section~\ref{sec:pagination}.
  \item Fields omitted from a \texttt{PATCH} body are left unchanged; send \texttt{null}
    explicitly to clear a field.
\end{itemize}

\section{Authentication}
Server-to-server integrations should use API keys; user-facing apps should use OAuth 2.0
bearer tokens. Both are sent via the \texttt{Authorization} header.

\begin{minipage}[t]{0.48\textwidth}
\begin{jsonpanel}{API KEY}{codebg}
\begin{lstlisting}
Authorization: ApiKey
  nb_live_7f3c9a2e1d4b8f0a
\end{lstlisting}
\end{jsonpanel}
\end{minipage}\hfill
\begin{minipage}[t]{0.48\textwidth}
\begin{jsonpanel}{OAUTH 2.0 BEARER}{codebg}
\begin{lstlisting}
Authorization: Bearer
  eyJhbGciOiJSUzI1NiIsInR5c...
\end{lstlisting}
\end{jsonpanel}
\end{minipage}
\par

{\small\color{steel} \infoicon{\faLock}\ API keys are generated per-workspace from
\textbf{Settings \textrightarrow\ Developer}. OAuth access tokens expire after 3{,}600
seconds; use the \texttt{refresh\_token} grant to renew without re-prompting the user.}

\section{Rate Limiting \& Errors}
Every response carries rate-limit headers, enforced per credential on a rolling one-hour
window. Exceeding the limit returns \texttt{429} with a \texttt{Retry-After} header.

\begin{center}
\begin{tabularx}{\linewidth}{@{}l l X@{}}
\toprule
\rowcolor{panelgray}
\textbf{Header} & \textbf{Example} & \textbf{Meaning} \\
\midrule
\texttt{X-RateLimit-Limit} & \texttt{1000} & Requests allowed in the current window \\
\texttt{X-RateLimit-Remaining} & \texttt{842} & Requests left before throttling \\
\texttt{X-RateLimit-Reset} & \texttt{1750939200} & Unix timestamp the window resets \\
\bottomrule
\end{tabularx}
\end{center}

Errors share one JSON shape across every endpoint:

\section{Files}
Endpoints for uploading, retrieving, updating, and removing files.

\subsection{List \& retrieve files}
\begin{routebox}
\httpbadge{GET}\hspace{6pt}\texttt{\textbf{/v2/files}}\hspace{10pt}
\httpbadge{GET}\hspace{6pt}\texttt{\textbf{/v2/files/\{file\_id\}}}\\[3pt]
{\small List files visible to the caller (most recently modified first), or fetch one
file's full metadata, including \texttt{download\_url} and \texttt{checksum\_sha256}.}
\end{routebox}

{\small\textbf{Query parameters (list only)}}
\begin{center}
\begin{tabularx}{\linewidth}{@{}p{2.3cm}p{1.3cm}p{1.7cm}X@{}}
\toprule
\rowcolor{panelgray}
\textbf{Name} & \textbf{Type} & \textbf{Required} & \textbf{Description} \\
\midrule
\texttt{folder\_id} & string & No & Restrict results to one folder \\
\texttt{limit} & integer & No & Page size, 1--100, default 20 \\
\texttt{cursor} & string & No & Opaque pagination cursor from a prior response \\
\bottomrule
\end{tabularx}
\end{center}

\begin{jsonpanel}{RESPONSE — 200 OK (single file object)}{respgreen}
\begin{lstlisting}
{
  "id": "fl_8a2c9f1e6d4b",
  "name": "q3-roadmap.pdf",
  "size_bytes": 482913,
  "mime_type": "application/pdf",
  "folder_id": "fd_2b7e1c9a",
  "modified_at": "2026-06-18T15:41:22Z"
}
\end{lstlisting}
\end{jsonpanel}

\subsection{Upload, update, and delete}
\begin{routebox}
\httpbadge{POST}\hspace{6pt}\texttt{\textbf{/v2/files}}\\[3pt]
{\small Creates a file record and returns a pre-signed upload URL. Send the raw bytes to
\texttt{upload\_url} within 15 minutes.}
\end{routebox}

\begin{minipage}[t]{0.48\textwidth}
\begin{jsonpanel}{REQUEST BODY}{reqblue}
\begin{lstlisting}
{
  "name": "q3-roadmap.pdf",
  "folder_id": "fd_2b7e1c9a",
  "mime_type": "application/pdf",
  "size_bytes": 482913
}
\end{lstlisting}
\end{jsonpanel}
\end{minipage}\hfill
\begin{minipage}[t]{0.48\textwidth}
\begin{jsonpanel}{RESPONSE — 201 CREATED}{respgreen}
\begin{lstlisting}
{
  "id": "fl_8a2c9f1e6d4b",
  "upload_url": "https://upload...",
  "expires_at": "2026-06-18T16:00Z"
}
\end{lstlisting}
\end{jsonpanel}
\end{minipage}
\par

\begin{routebox}
\httpbadge{PATCH}\hspace{6pt}\texttt{\textbf{/v2/files/\{file\_id\}}}\hspace{10pt}
\httpbadge{DELETE}\hspace{6pt}\texttt{\textbf{/v2/files/\{file\_id\}}}\\[3pt]
{\small \texttt{PATCH} renames or moves a file (\texttt{\{"name": ..., "folder\_id":
...\}}), returning \texttt{200} with the updated object. \texttt{DELETE} moves the file to
the trash (purged after 30 days); returns \texttt{204 No Content}. Pass
\texttt{?permanent=true} to skip the trash (requires \texttt{files:hard\_delete}).}
\end{routebox}

\section{Webhooks}
Subscribe to workspace events instead of polling. Nimbus signs every delivery with an
HMAC-SHA256 signature in the \texttt{X-Nimbus-Signature} header.

\begin{routebox}
\httpbadge{POST}\hspace{6pt}\texttt{\textbf{/v2/webhooks}}\hspace{10pt}
\httpbadge{DELETE}\hspace{6pt}\texttt{\textbf{/v2/webhooks/\{webhook\_id\}}}\\[3pt]
{\small Register an HTTPS endpoint and event list, or unregister one (\texttt{204 No
Content}).}
\end{routebox}

\begin{minipage}[t]{0.48\textwidth}
\begin{jsonpanel}{REQUEST BODY}{reqblue}
\begin{lstlisting}
{
  "url": "https://hooks.example-crm.io/nimbus",
  "events": ["file.created", "file.deleted"],
  "secret": "whsec_3f9a7c2e"
}
\end{lstlisting}
\end{jsonpanel}
\end{minipage}\hfill
\begin{minipage}[t]{0.48\textwidth}
\begin{jsonpanel}{DELIVERY — file.created}{codebg}
\begin{lstlisting}
{
  "event": "file.created",
  "occurred_at": "2026-06-18T15:41Z",
  "data": { "id": "fl_8a2c9f1e6d4b" }
}
\end{lstlisting}
\end{jsonpanel}
\end{minipage}

\section{Pagination}
\label{sec:pagination}
List endpoints use cursor-based pagination. Request the first page without a
\texttt{cursor}; if \texttt{next\_cursor} is present in the response, pass it back as the
\texttt{cursor} query parameter to fetch the next page. A response with no
\texttt{next\_cursor} field marks the end of the list.

\begin{jsonpanel}{REQUEST}{codebg}
\begin{lstlisting}
GET /v2/files?folder_id=fd_2b7e1c9a&limit=20&cursor=eyJvZmZzZXQiOjIwfQ==
\end{lstlisting}
\end{jsonpanel}
