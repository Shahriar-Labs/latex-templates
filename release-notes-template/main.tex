% ── Overview ──────────────────────────────────────────────────────────────
\begin{minipage}{\textwidth}
  \iconlabel{\faStar}{Overview}\\[4pt]
  \small
  Nimbus Platform v3.4.2 is a stability and quality-of-life release that
  delivers thirteen new features across the dashboard, API, and automation
  engine, resolves twenty-four reported defects, and formally deprecates two
  legacy endpoints ahead of the v4.0 migration in Q1 2027. All teams running
  v3.3.x or earlier should review the upgrade notes below before applying this
  update.
\end{minipage}

\vspace{10pt}

% ── NEW FEATURES ──────────────────────────────────────────────────────────
\section{\faStar\hspace{6pt}New Features}

\begin{featurebox}
  {\small\bfseries\color{accTeal}Dashboard \& Analytics\par}
  \vspace{4pt}

  \begin{itemize}[leftmargin=*,nosep,itemsep=0pt,topsep=4pt]
    \item \textbf{Custom Metric Builder.} Drag-and-drop interface for creating
      composite KPIs from any data source. Supports rolling windows,
      weighted averages, and conditional thresholds.
    \item \textbf{Shared Dashboard Templates.} Teams can now export dashboard
      layouts as JSON templates and import them across workspaces. Access
      control respects existing workspace roles.
    \item \textbf{Scheduled PDF Exports.} Dashboards can be configured to
      generate and email PDF snapshots on a cron schedule. Delivery targets
      support SMTP, webhook, and S3-compatible storage.
  \end{itemize}
\end{featurebox}

\begin{featurebox}
  {\small\bfseries\color{accTeal}API \& Developer Experience\par}
  \vspace{4pt}

  \begin{itemize}[leftmargin=*,nosep,itemsep=0pt,topsep=4pt]
    \item \textbf{Bulk Operation Endpoint.} New
      \texttt{POST /api/v3/bulk/execute} accepts up to 200 operations in a
      single request with atomic rollback on failure.
    \item \textbf{OpenAPI 3.1 Specification.} The full REST API surface is now
      documented in an OpenAPI 3.1 spec, available at
      \texttt{/api/v3/openapi.json}. SDK generation is supported via the
      Nimbus CLI.
    \item \textbf{Webhook Retry Policies.} Per-endpoint retry configuration
      with exponential backoff, jitter, and dead-letter queues. Delivery logs
      are retained for 30 days.
  \end{itemize}
\end{featurebox}

\begin{featurebox}
  {\small\bfseries\color{accTeal}Automation Engine\par}
  \vspace{4pt}

  \begin{itemize}[leftmargin=*,nosep,itemsep=0pt,topsep=4pt]
    \item \textbf{Conditional Branching.} Workflows now support
      \texttt{if/else} and \texttt{switch} nodes for routing based on runtime
      data. Evaluated server-side with sub-millisecond latency.
    \item \textbf{Parallel Execution Pools.} Designate resource pools for
      concurrent task execution with configurable max concurrency and priority
      queuing per tenant.
    \item \textbf{SLA Monitoring Built-in.} Each automation run now emits
      structured timing data to the observability pipeline. Pre-built Grafana
      dashboards included.
  \end{itemize}
\end{featurebox}

\begin{featurebox}
  {\small\bfseries\color{accTeal}Security\par}
  \vspace{4pt}

  \begin{itemize}[leftmargin=*,nosep,itemsep=0pt,topsep=4pt]
    \item \textbf{SCIM Provisioning (Preview).} System for Cross-domain
      Identity Management v2.0 support for automated user lifecycle
      management from Azure AD and Okta-compatible identity providers.
    \item \textbf{Audit Log Streaming.} Real-time streaming of all audit
      events to external SIEM platforms via Kafka or HTTP push. Configurable
      field-level redaction.
  \end{itemize}
\end{featurebox}

% ── BUG FIXES ─────────────────────────────────────────────────────────────
\section{\faBug\hspace{6pt}Resolved Issues}

\begin{bugbox}
  {\small\bfseries\color{accAmber}Critical Fixes\par}
  \vspace{4pt}

  \begin{itemize}[leftmargin=*,nosep,itemsep=0pt,topsep=4pt]
    \item \textbf{NB-4821} --- Race condition in the OAuth token refresh flow
      could cause concurrent requests to receive stale access tokens, resulting
      in transient 401 errors under high load. \textit{Fixed by: Marcus Webb.}
    \item \textbf{NB-4798} --- CSV exports exceeding 500,000 rows would
      silently truncate data without emitting a warning. Export service now
      chunks output and appends a completeness manifest.
    \item \textbf{NB-4760} --- Notification channels configured with
      non-ASCII display names caused publish failures on the Firebase Cloud
      Messaging adapter. Encoding is now normalized to UTF-8 at the adapter
      boundary.
  \end{itemize}
\end{bugbox}

\begin{bugbox}
  {\small\bfseries\color{accAmber}General Fixes\par}
  \vspace{4pt}

  \begin{itemize}[leftmargin=*,nosep,itemsep=0pt,topsep=4pt]
    \item \textbf{NB-4839} --- Timezone offset was not applied to
      \texttt{created\_at} timestamps in the activity feed when the user
      profile specified a non-UTC zone.
    \item \textbf{NB-4844} --- SAML assertion validation failed for IdPs
      that sign only the assertion element rather than the full response.
    \item \textbf{NB-4851} --- Search query highlighting stripped HTML
      entities from result snippets, occasionally producing garbled text.
    \item \textbf{NB-4867} --- The data connector for VertexDB returned
      stale schema caches after DDL changes, requiring a manual cache flush.
    \item \textbf{NB-4872} --- Pagination cursors on the user listing
      endpoint would reset when a filter parameter was cleared mid-session.
    \item \textbf{NB-4880} --- \texttt{NullPointerException} in the
      workflow scheduler when a task referenced a deleted integration
      credential. The scheduler now degrades gracefully with a clear error
      state.
  \end{itemize}
\end{bugbox}

% ── DEPRECATIONS ──────────────────────────────────────────────────────────
\section{\faExclamationTriangle\hspace{6pt}Deprecations \& Removals}

\begin{depbox}
  \small
  \begin{itemize}[leftmargin=*,nosep,itemsep=6pt,topsep=2pt]
    \item \textbf{\texttt{GET /api/v2/reports/generate}}
      \hfill\mbox{\releasebadge{End-of-Life Q1 2027}}\\[2pt]
      Replaced by \texttt{POST /api/v3/reports/schedule} with improved
      filtering, asynchronous generation, and webhook delivery. All v2 report
      integrations must migrate before March 1, 2027.

    \item \textbf{\texttt{POST /api/v2/webhooks/register}}
      \hfill\mbox{\releasebadge{End-of-Life Q1 2027}}\\[2pt]
      Superseded by the unified webhook management API at
      \texttt{/api/v3/webhooks}. The v2 endpoint no longer receives feature
      updates and will be removed in v4.0.

    \item \textbf{Legacy \texttt{X-Nimbus-Auth} Header}\\[2pt]
      The custom authentication header has been deprecated in favor of
      standard \texttt{Authorization: Bearer} tokens. Requests using the
      legacy header will receive a \texttt{Warning} response header starting
      with this release.
  \end{itemize}
\end{depbox}

\section{\faDownload\hspace{6pt}Upgrade Instructions}

\begin{releasebox}
  \small

  \textbf{Prerequisites.} v3.3.0 or later with PostgreSQL~15+ and Redis~7+.
  Backwards compatibility with PostgreSQL~14 is maintained but unsupported
  for new installations.

  \vspace{6pt}
  \textbf{Step 1 --- Backup.} Create a full database snapshot and export
  your workspace configurations via the Nimbus CLI:

  \texttt{\$ nimbus config export > workspace-backup.json}

  \vspace{6pt}
  \textbf{Step 2 --- Apply Migration.} Run the database migration script
  included in the release tarball:

  \texttt{\$ nimbus migrate apply -{}-release 3.4.2 -{}-dry-run}\\[2pt]
  \texttt{\$ nimbus migrate apply -{}-release 3.4.2}

  \vspace{6pt}
  \textbf{Step 3 --- Restart Services.} Restart all application nodes in
  rolling fashion. Zero-downtime upgrades are supported when running at least
  two nodes behind a load balancer.

  \vspace{6pt}
  \textbf{Rollback.} If issues arise, revert to v3.4.1 using the same
  migration tool with the \texttt{-{}-rollback} flag. Database schema changes
  in this release are backwards-compatible with v3.4.0+.

\end{releasebox}
