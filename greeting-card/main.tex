% ── Palette ─────────────────────────────────────────────────────────────────
\definecolor{cardbg}{HTML}{FDF6F0}   % warm cream
\definecolor{accent}{HTML}{8B3A62}   % plum

\pagecolor{cardbg}
\setlength{\parindent}{0pt}
\pagestyle{empty}

% ── A small floral / laurel ornament ────────────────────────────────────────
\newcommand{\ornament}[1][accent]{%
  \begin{tikzpicture}[scale=0.5]
    \draw[#1,line width=1pt] (-2,0) .. controls (-1,0.5) and (-0.4,0.2) .. (0,0)
                             .. controls (0.4,0.2) and (1,0.5) .. (2,0);
    \foreach \x in {-1.5,-1,-0.5,0.5,1,1.5}{
      \fill[gold] (\x,0) circle (1.6pt);}
    \fill[#1] (0,0) circle (2.4pt);
  \end{tikzpicture}}

% one panel = half the page width, full height
\newcommand{\panelW}{148.5mm}

\begin{document}

  {\color{accent}\Large\itshape Dear Friend,}\\[8mm]
