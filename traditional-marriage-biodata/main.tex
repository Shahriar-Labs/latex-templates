% Define the corner ornament command
\newcommand{\cornerornament}[1]{
    \begin{scope}[shift={#1}]
        % Outer decorative square
        \fill[deepred] (-12pt, -12pt) rectangle (12pt, 12pt);
        \draw[gold, line width=1.5pt] (-12pt, -12pt) rectangle (12pt, 12pt);
        % Inner circular flower motif
        \draw[gold, fill=deepred, line width=1pt] (0,0) circle (6pt);
        \draw[gold, fill=gold] (0,0) circle (2pt);
        \foreach \a in {45, 135, 225, 315} {
            \draw[gold, line width=1pt] (\a:6pt) -- (\a:12pt);
            \fill[gold] (\a:12pt) circle (1.5pt);
        }
    \end{scope}
}

% Define the lotus flower motif for headers/dividers
\newcommand{\lotusmotif}[1]{
    \begin{scope}[shift={#1}, scale=0.6]
        % Center petal
        \path[fill=deepred, draw=gold, line width=0.8pt] (0,0) 
            .. controls (-0.2,0.5) and (-0.2,1.2) .. (0,1.5) 
            .. controls (0.2,1.2) and (0.2,0.5) .. (0,0);
        % Inner side petals
        \path[fill=deepred, draw=gold, line width=0.8pt] (0,0.1) 
            .. controls (-0.5,0.4) and (-0.6,1.0) .. (-0.4,1.3) 
            .. controls (-0.2,1.0) and (-0.1,0.5) .. (0,0.1);
        \path[fill=deepred, draw=gold, line width=0.8pt] (0,0.1) 
            .. controls (0.5,0.4) and (0.6,1.0) .. (0.4,1.3) 
            .. controls (0.2,1.0) and (0.1,0.5) .. (0,0.1);
        % Outer side petals
        \path[fill=deepred, draw=gold, line width=0.8pt] (0,0.2) 
            .. controls (-0.8,0.2) and (-1.0,0.7) .. (-0.8,0.9) 
            .. controls (-0.5,0.7) and (-0.2,0.4) .. (0,0.2);
        \path[fill=deepred, draw=gold, line width=0.8pt] (0,0.2) 
            .. controls (0.8,0.2) and (1.0,0.7) .. (0.8,0.9) 
            .. controls (0.5,0.7) and (0.2,0.4) .. (0,0.2);
        % Gold base leaf
        \path[fill=gold, draw=deepred, line width=0.6pt] (-0.4,-0.1) 
            .. controls (-0.2,-0.2) and (0.2,-0.2) .. (0.4,-0.1) 
            -- (0.2,-0.25) -- (-0.2,-0.25) -- cycle;
    \end{scope}
}

% Section Header Command with Decorative Line
\newcommand{\biodatasection}[1]{
    \vspace{16pt}
    \noindent{\color{deepred}\Large\bfseries\scshape #1}
    \nopagebreak\\
    \noindent
    \begin{tikzpicture}
        \draw[deepred, line width=1.2pt] (0.6pt,0) -- (\linewidth-0.6pt,0);
        \draw[gold, line width=0.6pt] (0.3pt,-2.5pt) -- (\linewidth-0.3pt,-2.5pt);
        \path (0,0) -- (\linewidth,0) coordinate[midway] (M);
        \fill[deepred] ([yshift=-1.25pt]M) -- ++(4pt,4pt) -- ++(4pt,-4pt) -- ++(-4pt,-4pt) -- cycle;
        \fill[gold] ([yshift=-1.25pt,xshift=4pt]M) circle (1.5pt);
    \end{tikzpicture}
    \nopagebreak
    \vspace{8pt}
}
