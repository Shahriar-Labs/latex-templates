\section*{Overview}

The Nexa Platform API is a read-only GraphQL endpoint for querying projects, teams, tasks, and workspace metadata. All requests are served over HTTPS from a single endpoint. Authentication uses bearer tokens scoped to a workspace with the permissions of the issuing user.

\vspace{4pt}
\begin{schemabox}[title=Endpoint]
POST  https://api.nexaplatform.com/v2/graphql
\end{schemabox}

\section*{Core Types}

\subsection*{Project and Task}

\begin{schemabox}[title=Project \& Task Type Definitions]
type Project \{\\
\quad id:           ID!\\
\quad name:         String!\\
\quad slug:         String!\\
\quad description:  String\\
\quad team:         Team!\\
\quad tasks(\\
\qquad first:   Int\\
\qquad after:   String\\
\qquad status:  TaskStatus\\
\quad ):            TaskConnection!\\
\quad createdAt:    DateTime!\\
\quad updatedAt:    DateTime!\\
\quad archivedAt:   DateTime\\
\}\\[2pt]
enum TaskStatus \{ BACKLOG, TODO, IN\_PROGRESS, IN\_REVIEW, DONE, CANCELLED \}\\[2pt]
enum TaskPriority \{ P0, P1, P2, P3 \}\\[2pt]
type Task \{\\
\quad id:            ID!\\
\quad title:         String!\\
\quad description:   String\\
\quad status:        TaskStatus!\\
\quad priority:      TaskPriority!\\
\quad assignee:      User\\
\quad project:       Project!\\
\quad labels:        [String!]!\\
\quad dueDate:       Date\\
\quad createdAt:     DateTime!\\
\quad updatedAt:     DateTime!\\
\}
\end{schemabox}

\subsection*{User, Team, and Connections}

\begin{schemabox}[title=Supporting Types]
type User \{ id: ID!,  username: String!,  email: String!,  avatarUrl: String,  role: Role! \}\\[1pt]
enum Role \{ ADMIN, MEMBER, VIEWER \}\\[1pt]
type Team \{ id: ID!, name: String!, description: String, members(first: Int, after: String): UserConnection!, projects(first: Int, after: String): ProjectConnection!, createdAt: DateTime! \}\\[1pt]
type PageInfo \{ endCursor: String, hasNextPage: Boolean!, hasPreviousPage: Boolean!, startCursor: String \}\\[1pt]
type TaskConnection     \{ edges: [TaskEdge!]!,     pageInfo: PageInfo!, totalCount: Int! \}\\[1pt]
type TaskEdge           \{ node: Task!,     cursor: String! \}\\[1pt]
type UserConnection     \{ edges: [UserEdge!]!,     pageInfo: PageInfo!, totalCount: Int! \}\\[1pt]
type UserEdge           \{ node: User!,     cursor: String! \}\\[1pt]
type ProjectConnection  \{ edges: [ProjectEdge!]!,  pageInfo: PageInfo!, totalCount: Int! \}\\[1pt]
type ProjectEdge        \{ node: Project!,  cursor: String! \}
\end{schemabox}

\section*{Mutations}

Every mutation returns the affected object for single-round-trip reads. Input types follow the naming convention \texttt{*Input} with a corresponding \texttt{*Payload} return type.

\vspace{2pt}
\begin{schemabox}[title=Mutation Root \& Input Types (Partial)]
type Mutation \{\\
\quad createProject(input: CreateProjectInput!): CreateProjectPayload!\\
\quad updateProject(input: UpdateProjectInput!): UpdateProjectPayload!\\
\quad archiveProject(id: ID!): ArchiveProjectPayload!\\
\quad createTask(input: CreateTaskInput!): CreateTaskPayload!\\
\quad updateTask(input: UpdateTaskInput!): UpdateTaskPayload!\\
\quad setTaskStatus(id: ID!, status: TaskStatus!): SetTaskStatusPayload!\\
\}\\[2pt]
input CreateProjectInput \{ name: String!, teamId: ID!, description: String \}\\
type CreateProjectPayload \{ project: Project!, clientMutationId: String \}\\[2pt]
input CreateTaskInput \{ title: String!, projectId: ID!, status: TaskStatus, priority: TaskPriority, assigneeId: ID, labels: [String!], dueDate: Date \}\\
type CreateTaskPayload \{ task: Task!, clientMutationId: String \}\\[2pt]
input UpdateTaskInput \{ id: ID!, title: String, description: String, status: TaskStatus, priority: TaskPriority, assigneeId: ID, labels: [String!], dueDate: Date \}\\
type UpdateTaskPayload \{ task: Task!, clientMutationId: String \}
\end{schemabox}

\subsection*{Create a Task}

\vspace{2pt}
\begin{querybox}[title=Mutation: Create Task]
mutation CreateTask(\$input: CreateTaskInput!) \{\\
\quad createTask(input: \$input) \{\\
\qquad task \{ id, title, status, priority, project \{ slug \}, assignee \{ username \} \}\\
\qquad clientMutationId\\
\quad \}\\
\}\\[2pt]
Variables: \{ "input": \{ "title": "Deploy Vertex auth service to staging", "projectId": "UHJvamVjdC04MjE=", "priority": "P0", "labels": ["infra","auth"], "dueDate": "2026-08-04" \} \}
\end{querybox}

\vspace{4pt}

\begin{responsebox}[title=Response]
\{ "data": \{ "createTask": \{ "task": \{ "id": "VGFzay0zMTk=", "title": "Deploy Vertex auth service to staging", "status": "TODO", "priority": "P0", "project": \{ "slug": "meridian-migration" \}, "assignee": null \}, "clientMutationId": null \} \} \}
\end{responsebox}

\subsection*{Bulk Status Update with Aliases}

Use GraphQL aliases to run multiple mutations in one request.

\vspace{2pt}
\begin{querybox}[title=Mutation: Bulk Status Update via Aliases]
mutation BulkStatusUpdate \{\\
\quad t301: setTaskStatus(id: "VGFzay0zMDE=", status: DONE)       \{ task \{ id, status \} \}\\
\quad t302: setTaskStatus(id: "VGFzay0zMDI=", status: IN\_REVIEW)  \{ task \{ id, status \} \}\\
\quad t303: setTaskStatus(id: "VGFzay0zMDM=", status: IN\_PROGRESS) \{ task \{ id, status \} \}\\
\}
\end{querybox}
