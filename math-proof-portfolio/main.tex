\section{Number Theory}

\begin{Definition}[Rational and Irrational Numbers]
A real number $r$ is called \emph{rational} if it can be written as
$r = p/q$ where $p, q \in \mathbb{Z}$ and $q \neq 0$. Otherwise $r$ is
called \emph{irrational}.
\end{Definition}

\vspace{6pt}

\begin{Theorem}[Irrationality of $\sqrt{2}$]
The number $\sqrt{2}$ is irrational.
\end{Theorem}

\begin{proof}
Suppose for contradiction that $\sqrt{2}$ is rational. Then we can write
\[
  \sqrt{2} = \frac{p}{q}
\]
where $p, q \in \mathbb{Z}$, $q \neq 0$, and $\gcd(p, q) = 1$ (the
fraction is in lowest terms). Squaring both sides gives
\[
  2q^2 = p^2.
\]
Hence $p^2$ is even, which forces $p$ to be even (since if $p$ were odd,
$p^2$ would be odd). Write $p = 2k$ for some $k \in \mathbb{Z}$. Then
\[
  2q^2 = (2k)^2 = 4k^2 \implies q^2 = 2k^2,
\]
so $q^2$ is even, and again $q$ must be even. But then both $p$ and $q$
are divisible by $2$, contradicting $\gcd(p, q) = 1$.
Therefore $\sqrt{2}$ is irrational.
\end{proof}

\vspace{6pt}

\begin{Theorem}[Infinitude of Primes \normalfont(Euclid, $\sim$300\,\textsc{bc})]
There are infinitely many prime numbers.
\end{Theorem}

\begin{proof}
Suppose for contradiction that there are only finitely many primes,
say $p_1, p_2, \ldots, p_n$. Define
\[
  N = p_1 \cdot p_2 \cdots p_n + 1.
\]
Since $N > 1$, it has at least one prime divisor $p$. But for each
$i \in \{1, \ldots, n\}$, dividing $N$ by $p_i$ leaves remainder $1$,
so $p_i \nmid N$. Thus $p \notin \{p_1, \ldots, p_n\}$, contradicting
the assumption that the list was complete. Hence no finite list can
contain all primes.
\end{proof}

\section{Combinatorics and Induction}

\begin{Theorem}[Sum of First $n$ Positive Integers]
For every positive integer $n$,
\[
  \sum_{k=1}^{n} k \;=\; \frac{n(n+1)}{2}.
\]
\end{Theorem}

\begin{proof}
We proceed by \emph{mathematical induction} on $n$.

\pskip
\textbf{Base case} ($n = 1$):\quad
$\displaystyle\sum_{k=1}^{1} k = 1 = \frac{1 \cdot 2}{2}$. \checkmark

\pskip
\textbf{Inductive step}:\quad Assume the formula holds for some $n = m \geq 1$,
i.e.,
\[
  \sum_{k=1}^{m} k = \frac{m(m+1)}{2}.
\]
We must show it holds for $n = m + 1$. Indeed,
\begin{align*}
  \sum_{k=1}^{m+1} k
    &= \left(\sum_{k=1}^{m} k\right) + (m+1) \\[4pt]
    &= \frac{m(m+1)}{2} + (m+1) \\[4pt]
    &= (m+1)\!\left(\frac{m}{2} + 1\right) \\[4pt]
    &= \frac{(m+1)(m+2)}{2},
\end{align*}
which is the formula for $n = m + 1$. By the principle of mathematical
induction the result holds for all $n \in \mathbb{Z}^+$.
\end{proof}

\begin{Corollary}[Sum of First $n$ Odd Integers]
For every positive integer $n$,
\[
  \sum_{k=1}^{n}(2k-1) \;=\; n^2.
\]
\end{Corollary}

\begin{proof}
Using the theorem above,
\[
  \sum_{k=1}^{n}(2k-1)
    = 2\sum_{k=1}^{n}k - n
    = 2\cdot\frac{n(n+1)}{2} - n
    = n(n+1) - n
    = n^2. \qedhere
\]
\end{proof}

\section{Real Analysis}

\begin{Definition}[Limit of a Sequence]
A sequence $(a_n)_{n \geq 1}$ of real numbers \emph{converges} to $L \in
\mathbb{R}$, written $\lim_{n\to\infty} a_n = L$, if for every
$\varepsilon > 0$ there exists $N \in \mathbb{N}$ such that
\[
  n > N \implies |a_n - L| < \varepsilon.
\]
\end{Definition}

\vspace{6pt}

\begin{Proposition}[Convergence of a Geometric Sequence]
Let $r \in \mathbb{R}$ with $|r| < 1$. Then $\displaystyle\lim_{n \to \infty} r^n = 0$.
\end{Proposition}

\begin{proof}
If $r = 0$ the result is immediate. Assume $0 < |r| < 1$ and let
$\varepsilon > 0$ be given. We seek $N$ such that $n > N$ implies
$|r|^n < \varepsilon$.

Since $|r| < 1$, we have $1/|r| > 1$, so we may write $1/|r| = 1 + h$
for some $h > 0$. By Bernoulli's inequality,
\[
  \left(\frac{1}{|r|}\right)^{\!n} = (1+h)^n \geq 1 + nh \;>\; nh,
\]
hence
\[
  |r|^n = \frac{1}{(1+h)^n} < \frac{1}{nh}.
\]
Choose $N = \lceil 1/(h\varepsilon) \rceil$. Then for all $n > N$,
\[
  |r^n - 0| = |r|^n < \frac{1}{nh} < \frac{1}{Nh} \leq \varepsilon.
\]
Therefore $r^n \to 0$ as $n \to \infty$.
\end{proof}

\vspace{6pt}

\begin{Lemma}[Squeeze Lemma]
Let $(a_n)$, $(b_n)$, $(c_n)$ be sequences with $a_n \leq b_n \leq c_n$
for all sufficiently large $n$. If
$\lim_{n\to\infty} a_n = \lim_{n\to\infty} c_n = L$, then
$\lim_{n\to\infty} b_n = L$.
\end{Lemma}

\begin{proof}
Let $\varepsilon > 0$. Since $a_n \to L$ and $c_n \to L$, choose $N$
large enough that for all $n > N$:
\[
  L - \varepsilon < a_n \leq b_n \leq c_n < L + \varepsilon.
\]
Then $|b_n - L| < \varepsilon$ for all $n > N$, so $b_n \to L$.
\end{proof}
