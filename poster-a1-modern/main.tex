  \block{Introduction}{
    Drug--target interaction (DTI) prediction is a cornerstone of modern computer-aided drug discovery. By identifying potential therapeutic candidates computationally, researchers can prioritize compounds for wet-lab validation, reducing screening costs by over \textcolor{HighlightColor}{\textbf{80\%}}.
    
    \vspace{10pt}
    \textbf{Key Challenges in Existing Approaches:}
    \begin{itemize}[leftmargin=1.5em, itemsep=4pt, label=\textcolor{AccentColor}{\faAngleRight}]
      \item \textbf{Data Sparsity:} Known interactions cover only a tiny fraction of the chemical-protein space.
      \item \textbf{Modality Gaps:} Three-dimensional structures are unavailable for over 60\% of target proteins.
      \item \textbf{Generalization:} Models struggle with out-of-distribution drug classes and novel proteins.
    \end{itemize}
    
    \vspace{10pt}
    \textbf{Our Methodology (HeteroGDTI):}
    \begin{itemize}[leftmargin=1.5em, itemsep=4pt, label=\textcolor{AccentColor}{\faAngleRight}]
      \item We model DTI as a heterogeneous link prediction task on a multi-relational knowledge graph.
      \item A multi-layer heterogeneous graph transformer learns node embeddings from biochemical features.
      \item Cross-modal attention fuses sequence-based SMILES descriptors with protein language model representations.
    \end{itemize}
  }

  \block{Methods}{
    \textbf{Heterogeneous Graph Formulation} \\
    We construct a multi-relational graph $\mathcal{G} = (\mathcal{V}, \mathcal{E})$ comprising four node types: drugs ($\mathcal{V}_d$), proteins ($\mathcal{V}_p$), diseases ($\mathcal{V}_{ds}$), and pathways ($\mathcal{V}_{pw}$).
    
    \vspace{8pt}
    \textbf{Feature Encoding}
    \begin{itemize}[leftmargin=1.5em, itemsep=3pt, label=\textcolor{AccentColor}{\faAngleRight}]
      \item \textbf{Drugs:} 300-dimensional Morgan fingerprints combined with ChemBERTa transformer representations.
      \item \textbf{Proteins:} Pre-trained ESM-2 (650M parameter) protein language model embeddings (1280-d).
      \item \textbf{Diseases \& Pathways:} Node2Vec embeddings trained on the Unified Medical Language System (UMLS).
    \end{itemize}
    
    \vspace{8pt}
    \textbf{Heterogeneous Message Passing} \\
    The representation $\mathbf{h}_v^{(\ell)}$ of node $v$ at layer $\ell$ is computed by aggregating relation-specific messages:
    {\footnotesize
    \[
      \mathbf{h}_v^{(\ell)} = \sigma \Bigl( \mathbf{W}_{\phi(v)}^{(\ell)} \mathbf{h}_v^{(\ell-1)} + \sum_{r \in \mathcal{R}} \sum_{u \in \mathcal{N}_r(v)} \alpha_{u,v}^r \mathbf{W}_r^{(\ell)} \mathbf{h}_u^{(\ell-1)} \Bigr)
    \]
    }
    where $\mathcal{R}$ denotes relation types, and $\alpha_{u,v}^r$ is the relation-aware attention weight.

  \block{Conclusion}{
    \begin{itemize}[leftmargin=1.5em, itemsep=5pt, label=\textcolor{AccentColor}{\faAngleRight}]
      \item \textbf{HeteroGDTI} achieves state-of-the-art DTI prediction accuracy, outperforming existing models by a margin of up to \textcolor{HighlightColor}{\textbf{7.2\%}} in AUC-ROC.
      \item Fusing protein language model representations with SMILES embeddings via cross-modal attention is highly effective.
      \item The graph-based approach scales efficiently to accommodate large-scale genomic virtual screening pipelines.
    \end{itemize}
  }
