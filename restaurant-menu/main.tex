% Custom dot leader that aligns with text baseline
\newcommand{\menuleader}{%
  \nobreak\leaders\hbox to 0.45em{\hss\raisebox{0.15ex}{\tiny.}\hss}\hfill\nobreak
}

% Custom menu item command
\newcommand{\menuitem}[3]{%
  \noindent
  \begingroup
    {\large\bfseries\color{branddark} #1}%
    \menuleader
    {\large\bfseries\color{brandaccent} #2}%
  \endgroup\par
  \vspace{0.15em}
  \noindent
  {\itshape\small\color{branddark!80} #3}\par
  \vspace{0.75em}
}

% Custom section command
\newcommand{\menusection}[1]{%
  \nopagebreak
  \vspace{1.1em}
  \begin{center}
    {\color{brandaccent}\scshape\Large\textls[150]{#1}}
  \end{center}
  \vspace{0.5em}
}

\newcommand{\menuornament}{%

      \draw[color=brandaccent, line width=0.6pt] (-2.2,0) -- (-0.3,0);
      \draw[color=brandaccent, line width=0.6pt] (0.3,0) -- (2.2,0);
      \fill[color=brandaccent] (0,0.08) -- (0.12,0) -- (0,-0.08) -- (-0.12,0) -- cycle;
      \fill[color=brandaccent] (-0.2,0) circle (1.2pt);
      \fill[color=brandaccent] (0.2,0) circle (1.2pt);
    \end{tikzpicture}
  \end{center}
}
