\section{Mechanisms and Spread of Online Misinformation}

\bibentry
  {Vosoughi, S., Roy, D., \& Aral, S. (2018). The spread of true and false news online.
   \textit{Science}, \textit{359}(6380), 1146–1151.
   \href{https://doi.org/10.1126/science.aap9559}{https://doi.org/10.1126/science.aap9559}}
  {Drawing on a dataset of 126,000 rumour cascades spread by 3 million Twitter users over eleven
   years, Vosoughi et al.\ demonstrate that false news diffuses significantly faster, farther, and
   more broadly than accurate news — and that human behaviour, not bots, is the primary driver.
   \textit{Evaluation:} The study's large longitudinal dataset and rigorous statistical controls
   make it a landmark empirical contribution; the restriction to Twitter limits generalisation to
   other platforms. \textit{Relevance:} Provides indispensable quantitative evidence for arguments
   about misinformation's structural advantages over factual reporting in algorithmically curated
   environments.}

\bibentry
  {Pennycook, G., \& Rand, D. G. (2019). Lazy, not biased: Susceptibility to partisan fake news
   is better explained by lack of reasoning than by motivated reasoning.
   \textit{Cognition}, \textit{188}, 39–50.
   \href{https://doi.org/10.1016/j.cognition.2018.06.011}{https://doi.org/10.1016/j.cognition.2018.06.011}}
  {Using six experiments involving more than 3,400 U.S. participants, Pennycook and Rand challenge
   the dominant ``echo chamber'' narrative by showing that analytical thinking ability — not
   partisan motivation — predicts accurate discernment of fake-news headlines. \textit{Evaluation:}
   The experimental design is internally consistent, though U.S.-centric convenience samples
   constrain cross-cultural applicability. \textit{Relevance:} Shifts emphasis toward cognitive
   interventions (critical thinking education) rather than purely structural or regulatory solutions
   to misinformation.}

\section{Media Literacy: Interventions and Educational Frameworks}

\bibentry
  {Jeong, S.-H., Cho, H., \& Hwang, Y. (2012). Media literacy interventions: A meta-analytic
   review. \textit{Journal of Communication}, \textit{62}(3), 454–472.
   \href{https://doi.org/10.1111/j.1460-2466.2012.01643.x}{https://doi.org/10.1111/j.1460-2466.2012.01643.x}}
  {This meta-analysis synthesises 51 media literacy studies ($N > 7{,}000$) and reports
   consistently positive effects across age groups and media domains, with effect sizes ranging from
   small (knowledge gains) to moderate (attitude and behavioural change). \textit{Evaluation:}
   Rigorous meta-analytic methodology strengthens generalisability; however, most constituent
   studies predate social media's dominance. \textit{Relevance:} Establishes a robust empirical
   foundation for advocating media literacy curricula as a systemic response to misinformation,
   supporting policy recommendations in the paper's conclusion.}

\bibentry
  {Breakstone, J., Smith, M., Wineburg, S., Rapaport, A., Carle, J., Garland, M., \& Saavedra, A.
   (2021). Students' civic online reasoning: A national portrait.
   \textit{Educational Researcher}, \textit{50}(8), 505–515.
   \href{https://doi.org/10.3102/0013189X211017495}{https://doi.org/10.3102/0013189X211017495}}
  {A nationally representative survey of 3,446 U.S. high-school and college students reveals
   alarming gaps in civic online reasoning — the ability to evaluate the credibility of digital
   sources. Most students failed to identify sponsored content or investigate the provenance of
   viral images. \textit{Evaluation:} Stratified random sampling strengthens external validity;
   the study measures reasoning performance rather than self-reported competence, a methodological
   strength. \textit{Relevance:} Offers stark empirical grounding for the urgent need to
   institutionalise digital literacy training and informs the policy discussion in Section~4.}

\section{Platform Governance and Democratic Implications}

\bibentry
  {Wardle, C., \& Derakhshan, H. (2017). \textit{Information disorder: Toward an interdisciplinary
   framework for research and policy making}. Council of Europe.
   \href{https://rm.coe.int/information-disorder-report/168076277c}{https://rm.coe.int/information-disorder-report/168076277c}}
  {Wardle and Derakhshan present a typological framework distinguishing \textit{mis-information},
   \textit{dis-information}, and \textit{mal-information} based on the intent to harm and
   factual accuracy of content. The report synthesises communication scholarship with platform-policy
   analysis. \textit{Evaluation:} As a policy report rather than a peer-reviewed empirical study,
   it prioritises conceptual clarity over statistical evidence; nevertheless its taxonomy has been
   widely adopted by researchers and policymakers. \textit{Relevance:} The tripartite framework
   provides the definitional scaffolding for the paper's theoretical section and clarifies
   distinctions that are often conflated in public discourse.}

\bibentry
  {Roozenbeek, J., Schneider, C. R., Dryhurst, S., Kerr, J., Freeman, A. L. J., Recchia, G.,
   van der Bles, A. M., \& van der Linden, S. (2020). Susceptibility to misinformation about
   COVID-19 across 26 countries. \textit{Royal Society Open Science}, \textit{7}(10), 201199.
   \href{https://doi.org/10.1098/rsos.201199}{https://doi.org/10.1098/rsos.201199}}
  {Surveying 26,000 respondents across 26 countries during the early COVID-19 pandemic,
   Roozenbeek et al.\ find that analytical thinking, digital literacy, and institutional trust
   are the strongest predictors of misinformation resistance, while education alone is a weak
   predictor. \textit{Evaluation:} The multinational scale is exceptional; self-reported measures
   introduce social desirability bias. \textit{Relevance:} The cross-national scope enriches the
   paper with comparative evidence, countering purely U.S.-centric assumptions and supporting
   arguments for globally adaptable media literacy standards.}
